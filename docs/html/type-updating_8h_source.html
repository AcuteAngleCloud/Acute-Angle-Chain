24 // a class that tracks type dependencies between nodes, letting you
25 // update types efficiently when removing and altering code.
26 // altering code can alter types in the following way:
27 // * removing a break can make a block unreachable, if nothing else
28 // reaches it
29 // * altering the type of a child to unreachable can make the parent
30 // unreachable
31 struct TypeUpdater : public ExpressionStackWalker<TypeUpdater, UnifiedExpressionVisitor<TypeUpdater>> {
32  // Part 1: Scanning
33 
34  // track names to their blocks, so that when we remove a break to
35  // a block, we know how to find it if we need to update it
36  struct BlockInfo {
37  Block* block = nullptr;
38  int numBreaks = 0;
39  };
40  std::map<Name, BlockInfo> blockInfos;
41 
42  // track the parent of each node, as child type changes may lead to
43  // unreachability
44  std::map<Expression*, Expression*> parents;
45 
46  void visitExpression(Expression* curr) {
47  if (expressionStack.size() > 1) {
48  parents[curr] = expressionStack[expressionStack.size() - 2];
49  } else {
50  parents[curr] = nullptr; // this is the top level
51  }
52  // discover block/break relationships
53  if (auto* block = curr->dynCast<Block>()) {
54  if (block->name.is()) {
55  blockInfos[block->name].block = block;
56  }
57  } else if (auto* br = curr->dynCast<Break>()) {
58  // ensure info exists, discoverBreaks can then fill it
59  blockInfos[br->name];
60  } else if (auto* sw = curr->dynCast<Switch>()) {
61  // ensure info exists, discoverBreaks can then fill it
62  for (auto target : sw->targets) {
63  blockInfos[target];
64  }
65  blockInfos[sw->default_];
66  }
67  // add a break to the info, for break and switch
68  discoverBreaks(curr, +1);
69  }
70 
71  // Part 2: Updating
72 
73  // Node replacements, additions, removals and type changes should be noted. An
74  // exception is nodes you know will never be looked at again.
75 
76  // note the replacement of one node with another. this should be called
77  // after performing the replacement.
78  // this does *not* look into the node. you should do so yourself if necessary
79  void noteReplacement(Expression* from, Expression* to) {
80  auto parent = parents[from];
81  noteRemoval(from);
82  // if we are replacing with a child, i.e. a node that was already present
83  // in the ast, then we just have a type and parent to update
84  if (parents.find(to) != parents.end()) {
85  parents[to] = parent;
86  if (from->type != to->type) {
87  propagateTypesUp(to);
88  }
89  } else {
90  noteAddition(to, parent, from);
91  }
92  }
93 
94  // note the removal of a node
95  void noteRemoval(Expression* curr) {
96  noteRemovalOrAddition(curr, nullptr);
97  parents.erase(curr);
98  }

117  void noteAddition(Expression* curr, Expression* parent, Expression* previous = nullptr) {
118  assert(parents.find(curr) == parents.end()); // must not already exist
119  noteRemovalOrAddition(curr, parent);
120  // if we didn't replace with the exact same type, propagate types up
121  if (!(previous && previous->type == curr->type)) {
122  propagateTypesUp(curr);
123  }
124  }

126  // if parent is nullptr, this is a removal
127  void noteRemovalOrAddition(Expression* curr, Expression* parent) {
128  parents[curr] = parent;
129  discoverBreaks(curr, parent ? +1 : -1);
130  }

132  // adds (or removes) breaks depending on break/switch contents
133  void discoverBreaks(Expression* curr, int change) {
134  if (auto* br = curr->dynCast<Break>()) {
135  if (BranchUtils::isBranchTaken(br)) {
136  noteBreakChange(br->name, change, br->value);
137  }
138  } else if (auto* sw = curr->dynCast<Switch>()) {
139  if (BranchUtils::isBranchTaken(sw)) {
140  applySwitchChanges(sw, change);
141  }
142  }
143  }
144 
145  void applySwitchChanges(Switch* sw, int change) {
146  std::set<Name> seen;
147  for (auto target : sw->targets) {
148  if (seen.insert(target).second) {
149  noteBreakChange(target, change, sw->value);
150  }
151  }
152  if (seen.insert(sw->default_).second) {
153  noteBreakChange(sw->default_, change, sw->value);
154  }
155  }
156 
157  // note the addition of a node
158  void noteBreakChange(Name name, int change, Expression* value) {
159  auto iter = blockInfos.find(name);
160  if (iter == blockInfos.end()) {
161  return; // we can ignore breaks to loops
162  }
163  auto& info = iter->second;
164  info.numBreaks += change;
165  assert(info.numBreaks >= 0);
166  auto* block = info.block;
167  if (block) { // if to a loop, can ignore
168  if (info.numBreaks == 0) {
169  // dropped to 0! the block may now be unreachable. that
170  // requires that it doesn't have a fallthrough
171  makeBlockUnreachableIfNoFallThrough(block);
172  } else if (change == 1 && info.numBreaks == 1) {
173  // bumped to 1! the block may now be reachable
174  if (block->type != unreachable) {
175  return; // was already reachable, had a fallthrough
176  }
177  changeTypeTo(block, value ? value->type : none);
178  }
179  }
180  }

182  // alters the type of a node to a new type.
183  // this propagates the type change through all the parents.
184  void changeTypeTo(Expression* curr, WasmType newType) {
185  if (curr->type == newType) return; // nothing to do
186  curr->type = newType;
187  propagateTypesUp(curr);
188  }
189 
190  // given a node that has a new type, or is a new node, update
191  // all the parents accordingly. the existence of the node and
192  // any changes to it already occurred, this just updates the
193  // parents following that. i.e., nothing is done to the
194  // node we start on, it's done.
195  // the one thing we need to do here is propagate unreachability,
196  // no other change is possible
197  void propagateTypesUp(Expression* curr) {
198  if (curr->type != unreachable) return;
199  while (1) {
200  auto* child = curr;
201  curr = parents[child];
202  if (!curr) return;
203  // if a child of a break/switch is now unreachable, the
204  // break may no longer be taken. note that if we get here,
205  // this is an actually new unreachable child of the
206  // node, so if there is just 1 such child, it is us, and
207  // we are newly unreachable
208  if (auto* br = curr->dynCast<Break>()) {
209  int unreachableChildren = 0;
210  if (br->value && br->value->type == unreachable) unreachableChildren++;
211  if (br->condition && br->condition->type == unreachable) unreachableChildren++;
212  if (unreachableChildren == 1) {
213  // the break is no longer taken
214  noteBreakChange(br->name, -1, br->value);
215  }
216  } else if (auto* sw = curr->dynCast<Switch>()) {
217  int unreachableChildren = 0;
218  if (sw->value && sw->value->type == unreachable) unreachableChildren++;
219  if (sw->condition->type == unreachable) unreachableChildren++;
220  if (unreachableChildren == 1) {
221  applySwitchChanges(sw, -1);
222  }
223  }
224  // get ready to apply unreachability to this node
225  if (curr->type == unreachable) {
226  return; // already unreachable, stop here
227  }
228  // most nodes become unreachable if a child is unreachable,
229  // but exceptions exists
230  if (auto* block = curr->dynCast<Block>()) {
231  // if the block has breaks, it can keep its type
232  if (!block->name.is() || blockInfos[block->name].numBreaks == 0) {
233  curr->type = unreachable;
234  } else {
235  return; // did not turn
236  }
237  } else if (auto* iff = curr->dynCast<If>()) {
238  // may not be unreachable if just one side is
239  iff->finalize();
240  if (curr->type != unreachable) {
241  return; // did not turn
242  }
243  } else {
244  curr->type = unreachable;
245  }
246  }
247  }
248 
249  // efficiently update the type of a block, given the data we know. this
250  // can remove a concrete type and turn the block unreachable when it is
251  // unreachable, and it does this efficiently, without scanning the full
252  // contents
253  void maybeUpdateTypeToUnreachable(Block* curr) {
254  if (!isConcreteWasmType(curr->type)) {
255  return; // nothing concrete to change to unreachable
256  }
257  if (curr->name.is() && blockInfos[curr->name].numBreaks > 0) {
258  return;// has a break, not unreachable
259  }
260  // look for a fallthrough
261  makeBlockUnreachableIfNoFallThrough(curr);
262  }
263 
264  void makeBlockUnreachableIfNoFallThrough(Block* curr) {
265  if (curr->type == unreachable) {
266  return; // no change possible
267  }
268  for (auto* child : curr->list) {
269  if (child->type == unreachable) {
270  // no fallthrough, this block is now unreachable
271  changeTypeTo(curr, unreachable);
272  return;
273  }
274  }
275  }
276 };
