147  template<typename Value,Uptr maxBits>
148  FORCEINLINE void serializeVarInt(OutputStream& stream,Value& inValue,Value minValue,Value maxValue)
149  {
150  Value value = inValue;
151 
152  if(value < minValue || value > maxValue)
153  {
154  throw FatalSerializationException(std::string("out-of-range value: ") + std::to_string(minValue) + "<=" + std::to_string(value) + "<=" + std::to_string(maxValue));
155  }
156 
157  bool more = true;
158  while(more)
159  {
160  U8 outputByte = (U8)(value&127);
161  value >>= 7;
162  more = std::is_signed<Value>::value
163  ? (value != 0 && value != Value(-1)) || (value >= 0 && (outputByte & 0x40)) || (value < 0 && !(outputByte & 0x40))
164  : (value != 0);
165  if(more) { outputByte |= 0x80; }
166  *stream.advance(1) = outputByte;
167  };

170  template<typename Value,Uptr maxBits>
171  FORCEINLINE void serializeVarInt(InputStream& stream,Value& value,Value minValue,Value maxValue)
172  {
173  // First, read the variable number of input bytes into a fixed size buffer.
174  enum { maxBytes = (maxBits + 6) / 7 };
175  U8 bytes[maxBytes] = {0};
176  Uptr numBytes = 0;
177  I8 signExtendShift = (I8)sizeof(Value) * 8;
178  while(numBytes < maxBytes)
179  {
180  U8 byte = *stream.advance(1);
181  bytes[numBytes] = byte;
182  ++numBytes;
183  signExtendShift -= 7;
184  if(!(byte & 0x80)) { break; }
185  };
186 
187  // Ensure that the input does not encode more than maxBits of data.
188  enum { numUsedBitsInHighestByte = maxBits - (maxBytes-1) * 7 };
189  enum { highestByteUsedBitmask = U8(1<<numUsedBitsInHighestByte)-U8(1) };
190  enum { highestByteSignedBitmask = U8(~U8(highestByteUsedBitmask) & ~U8(0x80)) };
191  if((bytes[maxBytes-1] & ~highestByteUsedBitmask) != 0
192  && ((bytes[maxBytes-1] & ~highestByteUsedBitmask) != U8(highestByteSignedBitmask) || !std::is_signed<Value>::value))
193  { throw FatalSerializationException("Invalid LEB encoding: invalid final byte"); }
194 
195  // Decode the buffer's bytes into the output integer.
196  value = 0;
197  for(Uptr byteIndex = 0;byteIndex < maxBytes;++byteIndex)
198  { value |= Value(bytes[byteIndex] & ~0x80) << (byteIndex * 7); }
199 
200  // Sign extend the output integer to the full size of Value.
201  if(std::is_signed<Value>::value && signExtendShift > 0)
202  { value = Value(value << signExtendShift) >> signExtendShift; }
203 
204  // Check that the output integer is in the expected range.
205  if(value < minValue || value > maxValue)
206  { throw FatalSerializationException(std::string("out-of-range value: ") + std::to_string(minValue) + "<=" + std::to_string(value) + "<=" + std::to_string(maxValue)); }
207  }
