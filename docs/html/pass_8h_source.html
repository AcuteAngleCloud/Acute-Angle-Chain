156 class Pass {

160  // Override this to perform preparation work before the pass runs.
161  // This will be called before the pass is run on a module.
162  virtual void prepareToRun(PassRunner* runner, Module* module) {}
163 
164  // Implement this with code to run the pass on the whole module
165  virtual void run(PassRunner* runner, Module* module) {
166  WASM_UNREACHABLE();
167  }
168 
169  // Implement this with code to run the pass on a single function, for
170  // a function-parallel pass
171  virtual void runFunction(PassRunner* runner, Module* module, Function* function) {
172  WASM_UNREACHABLE();
173  }
174 
175  // Function parallelism. By default, passes are not run in parallel, but you
176  // can override this method to say that functions are parallelizable. This
177  // should always be safe *unless* you do something in the pass that makes it
178  // not thread-safe; in other words, the Module and Function objects and
179  // so forth are set up so that Functions can be processed in parallel, so
180  // if you do not ad global state that could be raced on, your pass could be
181  // function-parallel.
182  //
183  // Function-parallel passes create an instance of the Walker class per function.
184  // That means that you can't rely on Walker object properties to persist across
185  // your functions, and you can't expect a new object to be created for each
186  // function either (which could be very inefficient).
187  virtual bool isFunctionParallel() { return false; }
188 
189  // This method is used to create instances per function for a function-parallel
190  // pass. You may need to override this if you subclass a Walker, as otherwise
191  // this will create the parent class.
192  virtual Pass* create() { WASM_UNREACHABLE(); }
193 
194  std::string name;
195 
196 protected:
197  Pass() {}
198  Pass(Pass &) {}
199  Pass &operator=(const Pass&) = delete;
200 };
