43 // Appends extra to block, flattening out if extra is a block as well
44 void flattenAppend(Ref ast, Ref extra) {
45  int index;
46  if (ast[0] == BLOCK) index = 1;
47  else if (ast[0] == DEFUN) index = 3;
48  else abort();
49  if (extra[0] == BLOCK) {
50  for (size_t i = 0; i < extra[1]->size(); i++) {
51  ast[index]->push_back(extra[1][i]);
52  }
53  } else {
54  ast[index]->push_back(extra);
55  }
56 }

58 //
59 // Wasm2AsmBuilder - converts a WebAssembly module into asm.js
60 //
61 // In general, asm.js => wasm is very straightforward, as can
62 // be seen in asm2wasm.h. Just a single pass, plus a little
63 // state bookkeeping (breakStack, etc.), and a few after-the
64 // fact corrections for imports, etc. However, wasm => asm.js
65 // is tricky because wasm has statements == expressions, or in
66 // other words, things like `break` and `if` can show up
67 // in places where asm.js can't handle them, like inside an
68 // a loop's condition check.
69 //
70 // We therefore need the ability to lower an expression into
71 // a block of statements, and we keep statementizing until we
72 // reach a context in which we can emit those statments. This
73 // requires that we create temp variables to store values
74 // that would otherwise flow directly into their targets if
75 // we were an expression (e.g. if a loop's condition check
76 // is a bunch of statements, we execute those statements,
77 // then use the computed value in the loop's condition;
78 // we might also be able to avoid an assign to a temp var
79 // at the end of those statements, and put just that
80 // value in the loop's condition).
81 //
82 // It is possible to do this in a single pass, if we just
83 // allocate temp vars freely. However, pathological cases
84 // can easily show bad behavior here, with many unnecessary
85 // temp vars. We could rely on optimization passes like
86 // Emscripten's eliminate/registerize pair, but we want
87 // wasm2asm to be fairly fast to run, as it might run on
88 // the client.
89 //
90 // The approach taken here therefore performs 2 passes on
91 // each function. First, it finds which expression will need to
92 // be statementized. It also sees which labels can receive a break
93 // with a value. Given that information, in the second pass we can
94 // allocate // temp vars in an efficient manner, as we know when we
95 // need them and when their use is finished. They are allocated
96 // using an RAII class, so that they are automatically freed
97 // when the scope ends. This means that a node cannot allocate
98 // its own temp var; instead, the parent - which knows the
99 // child will return a value in a temp var - allocates it,
100 // and tells the child what temp var to emit to. The child
101 // can then pass forward that temp var to its children,
102 // optimizing away unnecessary forwarding.

128  IString ret;

198 Ref Wasm2AsmBuilder::processWasm(Module* wasm) {
199  Ref ret = ValueBuilder::makeToplevel();
200  Ref asmFunc = ValueBuilder::makeFunction(ASM_FUNC);
201  ret[1]->push_back(asmFunc);
202  ValueBuilder::appendArgumentToFunction(asmFunc, GLOBAL);
203  ValueBuilder::appendArgumentToFunction(asmFunc, ENV);
204  ValueBuilder::appendArgumentToFunction(asmFunc, BUFFER);
205  asmFunc[3]->push_back(ValueBuilder::makeStatement(ValueBuilder::makeString(USE_ASM)));
206  // create heaps, etc
207  addBasics(asmFunc[3]);
208  for (auto& import : wasm->imports) {
209  addImport(asmFunc[3], import.get());
210  }
211  // figure out the table size
212  tableSize = wasm->table.names.size();
213  size_t pow2ed = 1;
214  while (pow2ed < tableSize) {
215  pow2ed <<= 1;
216  }
217  tableSize = pow2ed;
218  // functions
219  for (auto& func : wasm->functions) {
220  asmFunc[3]->push_back(processFunction(func.get()));
221  }
222  addTables(asmFunc[3], wasm);
223  // memory XXX
224  addExports(asmFunc[3], wasm);
225  return ret;
226 }

288 void Wasm2AsmBuilder::addTables(Ref ast, Module *wasm) {
289  std::map<std::string, std::vector<IString>> tables; // asm.js tables, sig => contents of table
290  for (size_t i = 0; i < wasm->table.names.size(); i++) {
291  Name name = wasm->table.names[i];
292  auto func = wasm->getFunction(name);
293  std::string sig = getSig(func);
294  auto& table = tables[sig];
295  if (table.size() == 0) {
296  // fill it with the first of its type seen. we have to fill with something; and for asm2wasm output, the first is the null anyhow
297  table.resize(tableSize);
298  for (size_t j = 0; j < tableSize; j++) {
299  table[j] = fromName(name);
300  }
301  } else {
302  table[i] = fromName(name);
303  }
304  }
305  for (auto& pair : tables) {
306  auto& sig = pair.first;
307  auto& table = pair.second;
308  std::string stable = std::string("FUNCTION_TABLE_") + sig;
309  IString asmName = IString(stable.c_str(), false);
310  // add to asm module
311  Ref theVar = ValueBuilder::makeVar();
312  ast->push_back(theVar);
313  Ref theArray = ValueBuilder::makeArray();
314  ValueBuilder::appendToVar(theVar, asmName, theArray);
315  for (auto& name : table) {
316  ValueBuilder::appendToArray(theArray, ValueBuilder::makeName(name));
317  }
318  }
319 }

329 Ref Wasm2AsmBuilder::processFunction(Function* func) {
330  if (debug) std::cerr << " processFunction " << func->name << '\n';
331  Ref ret = ValueBuilder::makeFunction(fromName(func->name));
332  frees.clear();
333  frees.resize(std::max(i32, std::max(f32, f64)) + 1);
334  temps.clear();
335  temps.resize(std::max(i32, std::max(f32, f64)) + 1);
336  temps[i32] = temps[f32] = temps[f64] = 0;
337  // arguments
338  for (Index i = 0; i < func->getNumParams(); i++) {
339  IString name = fromName(func->getLocalName(i));
340  ValueBuilder::appendArgumentToFunction(ret, name);
341  ret[3]->push_back(
342  ValueBuilder::makeStatement(
343  ValueBuilder::makeAssign(
344  ValueBuilder::makeName(name),
345  makeAsmCoercion(ValueBuilder::makeName(name), wasmToAsmType(func->getLocalType(i)))
346  )
347  )
348  );
349  }
350  Ref theVar = ValueBuilder::makeVar();
351  size_t theVarIndex = ret[3]->size();
352  ret[3]->push_back(theVar);
353  // body
354  scanFunctionBody(func->body);
355  if (isStatement(func->body)) {
356  IString result = func->result != none ? getTemp(func->result) : NO_RESULT;
357  flattenAppend(ret, ValueBuilder::makeStatement(processFunctionBody(func, result)));
358  if (func->result != none) {
359  // do the actual return
360  ret[3]->push_back(ValueBuilder::makeStatement(ValueBuilder::makeReturn(makeAsmCoercion(ValueBuilder::makeName(result), wasmToAsmType(func->result)))));
361  freeTemp(func->result, result);
362  }
363  } else {
364  // whole thing is an expression, just do a return
365  if (func->result != none) {
366  ret[3]->push_back(ValueBuilder::makeStatement(ValueBuilder::makeReturn(makeAsmCoercion(processFunctionBody(func, EXPRESSION_RESULT), wasmToAsmType(func->result)))));
367  } else {
368  flattenAppend(ret, processFunctionBody(func, NO_RESULT));
369  }
370  }
371  // vars, including new temp vars
372  for (Index i = func->getVarIndexBase(); i < func->getNumLocals(); i++) {
373  ValueBuilder::appendToVar(theVar, fromName(func->getLocalName(i)), makeAsmCoercedZero(wasmToAsmType(func->getLocalType(i))));
374  }
375  for (auto f : frees[i32]) {
376  ValueBuilder::appendToVar(theVar, f, makeAsmCoercedZero(ASM_INT));
377  }
378  for (auto f : frees[f32]) {
379  ValueBuilder::appendToVar(theVar, f, makeAsmCoercedZero(ASM_FLOAT));
380  }
381  for (auto f : frees[f64]) {
382  ValueBuilder::appendToVar(theVar, f, makeAsmCoercedZero(ASM_DOUBLE));
383  }
384  if (theVar[1]->size() == 0) {
385  ret[3]->splice(theVarIndex, 1);
386  }
387  // checks
388  assert(frees[i32].size() == temps[i32]); // all temp vars should be free at the end
389  assert(frees[f32].size() == temps[f32]); // all temp vars should be free at the end
390  assert(frees[f64].size() == temps[f64]); // all temp vars should be free at the end
391  // cleanups
392  willBeStatement.clear();
393  return ret;
394 }

500  // A scoped temporary variable.
501  struct ScopedTemp {
502  Wasm2AsmBuilder* parent;
503  WasmType type;
504  IString temp;
505  bool needFree;
506  // @param possible if provided, this is a variable we can use as our temp. it has already been
507  // allocated in a higher scope, and we can just assign to it as our result is
508  // going there anyhow.
509  ScopedTemp(WasmType type, Wasm2AsmBuilder* parent, IString possible = NO_RESULT) : parent(parent), type(type) {
510  assert(possible != EXPRESSION_RESULT);
511  if (possible == NO_RESULT) {
512  temp = parent->getTemp(type);
513  needFree = true;
514  } else {
515  temp = possible;
516  needFree = false;
517  }
518  }
519  ~ScopedTemp() {
520  if (needFree) {
521  parent->freeTemp(type, temp);
522  }
523  }
524 
525  IString getName() {
526  return temp;
527  }
528  Ref getAstName() {
529  return ValueBuilder::makeName(temp);
530  }
531  };

533  Ref visit(Expression* curr, IString nextResult) {
534  IString old = result;
535  result = nextResult;
536  Ref ret = Visitor::visit(curr);
537  result = old; // keep it consistent for the rest of this frame, which may call visit on multiple children
538  return ret;
539  }
540 
541  Ref visit(Expression* curr, ScopedTemp& temp) {
542  return visit(curr, temp.temp);
543  }

545  Ref visitForExpression(Expression* curr, WasmType type, IString& tempName) { // this result is for an asm expression slot, but it might be a statement
546  if (isStatement(curr)) {
547  ScopedTemp temp(type, parent);
548  tempName = temp.temp;
549  return visit(curr, temp);
550  } else {
551  return visit(curr, EXPRESSION_RESULT);
552  }
553  }

555  Ref visitAndAssign(Expression* curr, IString result) {
556  Ref ret = visit(curr, result);
557  // if it's not already a statement, then it's an expression, and we need to assign it
558  // (if it is a statement, it already assigns to the result var)
559  if (!isStatement(curr) && result != NO_RESULT) {
560  ret = ValueBuilder::makeStatement(ValueBuilder::makeAssign(ValueBuilder::makeName(result), ret));
561  }
562  return ret;
563  }

565  Ref visitAndAssign(Expression* curr, ScopedTemp& temp) {
566  return visitAndAssign(curr, temp.getName());
567  }

573  // Expressions with control flow turn into a block, which we must
574  // then handle, even if we are an expression.
575  bool isBlock(Ref ast) {
576  return !!ast && ast[0] == BLOCK;

579  Ref blockify(Ref ast) {
580  if (isBlock(ast)) return ast;
581  Ref ret = ValueBuilder::makeBlock();
582  ret[1]->push_back(ValueBuilder::makeStatement(ast));
583  return ret;
584  }

586  // For spooky return-at-a-distance/break-with-result, this tells us
587  // what the result var is for a specific label.
588  std::map<Name, IString> breakResults;
589 
590  // Breaks to the top of a loop should be emitted as continues, to that loop's main label
591  std::map<Name, Name> continueLabels;
592 
593  IString fromName(Name name) {
594  return parent->fromName(name);
595  }
596 
597  // Visitors
598 
599  Ref visitBlock(Block *curr) {
600  breakResults[curr->name] = result;
601  Ref ret = ValueBuilder::makeBlock();
602  size_t size = curr->list.size();
603  auto noResults = result == NO_RESULT ? size : size-1;
604  for (size_t i = 0; i < noResults; i++) {
605  flattenAppend(ret, ValueBuilder::makeStatement(visit(curr->list[i], NO_RESULT)));
606  }
607  if (result != NO_RESULT) {
608  flattenAppend(ret, visitAndAssign(curr->list[size-1], result));
609  }
610  if (curr->name.is()) {
611  ret = ValueBuilder::makeLabel(fromName(curr->name), ret);
612  }
613  return ret;
614  }
615  Ref visitIf(If *curr) {
616  IString temp;
617  Ref condition = visitForExpression(curr->condition, i32, temp);
618  Ref ifTrue = ValueBuilder::makeStatement(visitAndAssign(curr->ifTrue, result));
619  Ref ifFalse;
620  if (curr->ifFalse) {
621  ifFalse = ValueBuilder::makeStatement(visitAndAssign(curr->ifFalse, result));
622  }
623  if (temp.isNull()) {
624  return ValueBuilder::makeIf(condition, ifTrue, ifFalse); // simple if
625  }
626  condition = blockify(condition);
627  // just add an if to the block
628  condition[1]->push_back(ValueBuilder::makeIf(ValueBuilder::makeName(temp), ifTrue, ifFalse));
629  return condition;
630  }
631  Ref visitLoop(Loop *curr) {
632  Name asmLabel = curr->out.is() ? curr->out : curr->in; // label using the outside, normal for breaks. if no outside, then inside
633  if (curr->in.is()) continueLabels[curr->in] = asmLabel;
634  Ref body = visit(curr->body, result);
635  Ref ret = ValueBuilder::makeDo(body, ValueBuilder::makeInt(0));
636  if (asmLabel.is()) {
637  ret = ValueBuilder::makeLabel(fromName(asmLabel), ret);
638  }
639  return ret;
640  }
641  Ref visitBreak(Break *curr) {
642  if (curr->condition) {
643  // we need an equivalent to an if here, so use that code
644  Break fakeBreak = *curr;
645  fakeBreak.condition = nullptr;
646  If fakeIf(allocator);
647  fakeIf.condition = curr->condition;
