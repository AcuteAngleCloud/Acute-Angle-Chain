121 #define CONSTRUCT_OP_HAS_DATA( r, DATA, OP ) \
122 template <typename ... Mutators> \
123 struct OP final : instr_base<Mutators...> { \
124  uint16_t code = BOOST_PP_CAT(OP,_code); \
125  DATA field; \
126  uint16_t get_code() override { return BOOST_PP_CAT(OP,_code); } \
127  int skip_ahead() override { return field_specific_params<DATA>::skip_ahead; } \
128  void unpack( char* opcode ) override { \
129  field_specific_params<DATA>::unpack( opcode, field ); \
130  } \
131  std::vector<U8> pack() override { \
132  std::vector<U8> output = { U8(code), 0 }; \
133  std::vector<U8> field_pack = field_specific_params<DATA>::pack( field ); \
134  output.insert( output.end(), field_pack.begin(), field_pack.end() ); \
135  return output; \
136  } \
137  std::string to_string() override { \
138  return std::string(BOOST_PP_STRINGIZE(OP))+field_specific_params<DATA>::to_string( field ); \
139  } \

652 template <class Op_Types>
653 struct AACIO_OperatorDecoderStream
654 {
655  AACIO_OperatorDecoderStream(const std::vector<U8>& codeBytes)
656  : start(codeBytes.data()), nextByte(codeBytes.data()), end(codeBytes.data()+codeBytes.size()) {
657  if(!_cached_ops)
658  _cached_ops = cached_ops<Op_Types>::get_cached_ops();
659  }
660 
661  operator bool() const { return nextByte < end; }
662 
663  instr* decodeOp() {
664  FC_ASSERT(nextByte + sizeof(IR::Opcode) <= end);
665  IR::Opcode opcode = *(IR::Opcode*)nextByte;
666  switch(opcode)
667  {
668  #define VISIT_OPCODE(opcode,name,nameString,Imm,...) \
669  case IR::Opcode::name: \
670  { \
671  FC_ASSERT(nextByte + sizeof(IR::OpcodeAndImm<IR::Imm>) <= end); \
672  IR::OpcodeAndImm<IR::Imm>* encodedOperator = (IR::OpcodeAndImm<IR::Imm>*)nextByte; \
673  nextByte += sizeof(IR::OpcodeAndImm<IR::Imm>); \
674  auto op = _cached_ops->at(BOOST_PP_CAT(name, _code)); \
675  op->unpack( reinterpret_cast<char*>(&(encodedOperator->imm)) ); \
676  return op; \
677  }
678  ENUM_OPERATORS(VISIT_OPCODE)
679  #undef VISIT_OPCODE
680  default:
681  nextByte += sizeof(IR::Opcode);
682  return _cached_ops->at(error_code);
683  }
684  }
