57  // Check if an expression is a sign-extend, and if so, returns the value
58  // that is extended, otherwise nullptr
59  static Expression* getSignExtValue(Expression* curr) {
60  if (auto* outer = curr->dynCast<Binary>()) {
61  if (outer->op == ShrSInt32) {
62  if (auto* outerConst = outer->right->dynCast<Const>()) {
63  if (auto* inner = outer->left->dynCast<Binary>()) {
64  if (inner->op == ShlInt32) {
65  if (auto* innerConst = inner->right->dynCast<Const>()) {
66  if (outerConst->value == innerConst->value) {
67  return inner->left;
68  }
69  }
70  }
71  }
72  }
73  }
74  }
75  return nullptr;
76  }

78  // gets the size of the sign-extended value
79  static Index getSignExtBits(Expression* curr) {
80  return 32 - curr->cast<Binary>()->right->cast<Const>()->value.geti32();
81  }

83  // Check if an expression is almost a sign-extend: perhaps the inner shift
84  // is too large. We can split the shifts in that case, which is sometimes
85  // useful (e.g. if we can remove the signext)
86  static Expression* getAlmostSignExt(Expression* curr) {
87  if (auto* outer = curr->dynCast<Binary>()) {
88  if (outer->op == ShrSInt32) {
89  if (auto* outerConst = outer->right->dynCast<Const>()) {
90  if (auto* inner = outer->left->dynCast<Binary>()) {
91  if (inner->op == ShlInt32) {
92  if (auto* innerConst = inner->right->dynCast<Const>()) {
93  if (outerConst->value.leU(innerConst->value).geti32()) {
94  return inner->left;
95  }
96  }
97  }
98  }
99  }
100  }
101  }
102  return nullptr;
103  }

105  // gets the size of the almost sign-extended value, as well as the
106  // extra shifts, if any
107  static Index getAlmostSignExtBits(Expression* curr, Index& extraShifts) {
108  extraShifts = curr->cast<Binary>()->left->cast<Binary>()->right->cast<Const>()->value.geti32() -
109  curr->cast<Binary>()->right->cast<Const>()->value.geti32();
110  return getSignExtBits(curr);
111  }

113  // Check if an expression is a zero-extend, and if so, returns the value
114  // that is extended, otherwise nullptr
115  static Expression* getZeroExtValue(Expression* curr) {
116  if (auto* binary = curr->dynCast<Binary>()) {
117  if (binary->op == AndInt32) {
118  if (auto* c = binary->right->dynCast<Const>()) {
119  if (Bits::getMaskedBits(c->value.geti32())) {
120  return binary->right;
121  }
122  }
123  }
124  }
125  return nullptr;
126  }
127 
128  // gets the size of the sign-extended value
129  static Index getZeroExtBits(Expression* curr) {
130  return Bits::getMaskedBits(curr->cast<Binary>()->right->cast<Const>()->value.geti32());
131  }
