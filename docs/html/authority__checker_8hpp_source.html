135  bool satisfied(const AuthorityType& authority, uint16_t depth = 0) {
136  // This check is redundant, since weight_tally_visitor did it too, but I'll leave it here for future-proofing
137  if (depth > recursion_depth_limit)

140  // Save the current used keys; if we do not satisfy this authority, the newly used keys aren't actually used
141  auto KeyReverter = fc::make_scoped_exit([this, keys = _used_keys] () mutable {
142  _used_keys = keys;
143  });

148  permissions.insert(authority.keys.begin(), authority.keys.end());

151  // Check all permissions, from highest weight to lowest, seeing if signing_keys satisfies them or not
152  weight_tally_visitor visitor(*this, depth);

154  // If we've got enough weight, to satisfy the authority, return!
