30  template <template <std::size_t, std::size_t> class Visitor, class T, class U>
31  bool binary_visit(const T& x, const U& y) {
32  constexpr std::size_t fields_count_lhs = detail::fields_count<std::remove_reference_t<T>>();
33  constexpr std::size_t fields_count_rhs = detail::fields_count<std::remove_reference_t<U>>();
34  constexpr std::size_t fields_count_min = detail::min_size(fields_count_lhs, fields_count_rhs);
35  typedef Visitor<0, fields_count_min> visitor_t;
36 
37 #if BOOST_PFR_USE_CPP17
38  return visitor_t::cmp(detail::tie_as_tuple(x), detail::tie_as_tuple(y));
39 #else
40  bool result = true;
41  ::boost::pfr::detail::for_each_field_dispatcher(
42  x,
43  [&result, &y](const auto& lhs) {
44  ::boost::pfr::detail::for_each_field_dispatcher(
45  y,
46  [&result, &lhs](const auto& rhs) {
47  result = visitor_t::cmp(lhs, rhs);
48  },
49  std::make_index_sequence<fields_count_rhs>{}
50  );
51  },
52  std::make_index_sequence<fields_count_lhs>{}
53  );
54 
55  return result;
