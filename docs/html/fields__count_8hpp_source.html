50 // Hand-made is_aggregate<T> trait:
51 // Aggregates could be constructed from `decltype(ubiq_constructor{I})...` but report that there's no constructor from `decltype(ubiq_constructor{I})...`
52 // Special case for N == 1: `std::is_constructible<T, ubiq_constructor>` returns true if N == 1 and T is copy/move constructible.
53 template <class T, std::size_t N>
54 struct is_aggregate_initializable_n {
55  template <std::size_t ...I>
56  static constexpr bool is_not_constructible_n(std::index_sequence<I...>) noexcept {
57  return !std::is_constructible<T, decltype(ubiq_constructor{I})...>::value
58  || is_single_field_and_aggregate_initializable<N, T>::value
59  ;
60  }
61 
62  static constexpr bool value =
63  std::is_empty<T>::value
64  || std::is_fundamental<T>::value
65  || is_not_constructible_n(std::make_index_sequence<N>{})

118 template <class T, std::size_t Begin, std::size_t Last>
119 constexpr std::size_t detect_fields_count_greedy(size_t_<Begin>, size_t_<Last>) noexcept {
120  constexpr std::size_t middle = Begin + (Last - Begin) / 2;
121  constexpr std::size_t fields_count_big = detect_fields_count_greedy<T>(size_t_<middle + 1>{}, size_t_<Last>{});
122  constexpr std::size_t fields_count_small = detect_fields_count_greedy<T>(size_t_<Begin>{}, size_t_<fields_count_big ? Begin : middle>{});
123  return fields_count_big ? fields_count_big : fields_count_small;

134 template <class T, std::size_t N>
135 constexpr std::size_t detect_fields_count_dispatch(size_t_<N>, int) noexcept {
136  // T is not default aggregate initialzable. It means that at least one of the members is not default constructible,
137  // so we have to check all the aggregate initializations for T up to N parameters and return the bigest succeeded
138  // (we can not use binary search for detecting fields count).
139  return detect_fields_count_greedy<T>(size_t_<0>{}, size_t_<N>{});
140 }

147  static_assert(

166  "Type must be aggregate initializable."

174 // "Type must be aggregate initializable."
175 // );

178  constexpr std::size_t max_fields_count = (sizeof(type) * 8); // We multiply by 8 because we may have bitfields in T
