32 //
33 // OptimizingIncrementalModuleBuilder
34 //
35 // Helps build wasm modules efficiently. If you build a module by
36 // adding function by function, and you want to optimize them, this class
37 // starts optimizing using worker threads *while you are still adding*.
38 // It runs function optimization passes at that time, and then at the end
39 // it runs global module-level optimization passes. The result is a fully
40 // optimized module, optimized while being generated.
41 //
42 // This might also be faster than normal module optimization since it
43 // runs all passes on each function, then goes on to the next function
44 // which is better for data locality.
45 //
46 // Usage: Create an instance, passing it the module and the total
47 // number of functions. Then call addFunction as you have
48 // new functions to add (this also adds it to the module). Finally,
49 // call finish() when all functions have been added.
50 //
51 // This avoids locking by using atomics. We allocate an array of nullptrs
52 // that represent all the functions, and as we add a function, we place it
53 // at the next index. Each worker will read from the start to the end,
54 // and when a non-nullptr is found, the worker optimizes that function and
55 // nulls it. There is also an end marker that is not nullptr nor the address of
56 // a valid function, which represents that beyond this point we have not
57 // yet filled in. In other words,
58 // * the main thread fills everything with the end marker
59 // * the main thread transforms a marker entry into a function
60 // * workers pause when they see the marker
61 // * workers skip over nullptrs
62 // * workers transform functions into nullptrs, and optimize them
63 // * we keep an atomic count of the number of active workers and
64 // the number of optimized functions.
65 // * after adding a function, the main thread wakes up workers if
66 // it calculates there is work for them.
67 // * a lock is used for going to sleep and waking up.
68 // Locking should be rare, as optimization is
69 // generally slower than generation; in the optimal case, we never
70 // lock beyond the first step, and all further work is lock-free.
71 //
72 
73 class OptimizingIncrementalModuleBuilder {
74  Module* wasm;
75  uint32_t numFunctions;
76  PassOptions passOptions;
77  std::function<void (PassRunner&)> addPrePasses;
78  Function* endMarker;
79  std::atomic<Function*>* list;
80  uint32_t nextFunction; // only used on main thread
81  uint32_t numWorkers;
82  std::vector<std::unique_ptr<std::thread>> threads;
83  std::atomic<uint32_t> liveWorkers, activeWorkers, availableFuncs, finishedFuncs;
84  std::mutex mutex;
85  std::condition_variable condition;

87  bool debug;
88  bool validateGlobally;

90 public:
91  // numFunctions must be equal to the number of functions allocated, or higher. Knowing
92  // this bounds helps avoid locking.
93  OptimizingIncrementalModuleBuilder(Module* wasm, Index numFunctions, PassOptions passOptions, std::function<void (PassRunner&)> addPrePasses, bool debug, bool validateGlobally)
94  : wasm(wasm), numFunctions(numFunctions), passOptions(passOptions), addPrePasses(addPrePasses), endMarker(nullptr), list(nullptr), nextFunction(0),
95  numWorkers(0), liveWorkers(0), activeWorkers(0), availableFuncs(0), finishedFuncs(0),
96  finishing(false), debug(debug), validateGlobally(validateGlobally) {
97 
98  if (!useWorkers()) {
99  // if we shouldn't use threads, don't
100  return;
101  }
102 
103  // Before parallelism, create all passes on the main thread here, to ensure
104  // prepareToRun() is called for each pass before we start to optimize functions.
105  {
106  PassRunner passRunner(wasm, passOptions);
107  addPrePasses(passRunner);
108  passRunner.addDefaultFunctionOptimizationPasses();
109  }
110 
111  // prepare work list
112  endMarker = new Function();
113  list = new std::atomic<Function*>[numFunctions];
114  for (uint32_t i = 0; i < numFunctions; i++) {
115  list[i].store(endMarker);
116  }
117  // create workers
118  DEBUG_THREAD("creating workers");
119  numWorkers = ThreadPool::getNumCores();
120  assert(numWorkers >= 1);
121  // worth it to use threads
122  liveWorkers.store(0);
123  activeWorkers.store(0);
124  for (uint32_t i = 0; i < numWorkers; i++) { // TODO: one less, and add it at the very end, to not compete with main thread?
125  createWorker();
126  }
127  waitUntilAllReady();
128  DEBUG_THREAD("workers are ready");
129  // prepare the rest of the initial state
130  availableFuncs.store(0);
131  finishedFuncs.store(0);
132  }
133 
134  ~OptimizingIncrementalModuleBuilder() {
135  delete[] list;
136  delete endMarker;
137  }
138 
139  bool useWorkers() {
140  return numFunctions > 0 && !debug && ThreadPool::getNumCores() > 1 && !PassRunner::getPassDebug();
141  }
142 
143  // Add a function to the module, and to be optimized
144  void addFunction(Function* func) {
145  wasm->addFunction(func);
146  if (!useWorkers()) return; // we optimize at the end in that case
147  queueFunction(func);
148  // wake workers if needed
149  auto wake = availableFuncs.load();
150  for (uint32_t i = 0; i < wake; i++) {
151  wakeWorker();
152  }
153  }
154 
155  // All functions have been added, block until all are optimized, and then do
156  // global optimizations. When this returns, the module is ready and optimized.
157  void finish() {
158  if (!useWorkers()) {
159  // optimize each function now that we are done adding functions,
160  // then optimize globally
161  PassRunner passRunner(wasm, passOptions);
162  if (debug) {
163  passRunner.setDebug(true);
164  passRunner.setValidateGlobally(validateGlobally);
165  }
166  addPrePasses(passRunner);
167  passRunner.addDefaultFunctionOptimizationPasses();
168  passRunner.addDefaultGlobalOptimizationPasses();
169  passRunner.run();
170  return;
171  }
172  DEBUG_THREAD("finish()ing");
173  assert(nextFunction == numFunctions);
174  wakeAllWorkers();
175  waitUntilAllFinished();
176  optimizeGlobally();
177  // TODO: clear side thread allocators from module allocator, as these threads were transient
178  }

180 private:
181  void createWorker() {
182  DEBUG_THREAD("create a worker");
183  threads.emplace_back(make_unique<std::thread>(workerMain, this));
184  }
185 
186  void wakeWorker() {
187  DEBUG_THREAD("wake a worker");
188  std::lock_guard<std::mutex> lock(mutex);
189  condition.notify_one();
190  }
191 
192  void wakeAllWorkers() {
193  DEBUG_THREAD("wake all workers");
194  std::lock_guard<std::mutex> lock(mutex);
195  condition.notify_all();
196  }
197 
198  void waitUntilAllReady() {
199  DEBUG_THREAD("wait until all workers are ready");
200  std::unique_lock<std::mutex> lock(mutex);
201  if (liveWorkers.load() < numWorkers) {
202  condition.wait(lock, [this]() { return liveWorkers.load() == numWorkers; });
203  }
204  }
205 
206  void waitUntilAllFinished() {
207  DEBUG_THREAD("wait until all workers are finished");
208  {
209  std::unique_lock<std::mutex> lock(mutex);
210  finishing = true;
211  if (liveWorkers.load() > 0) {
212  condition.wait(lock, [this]() { return liveWorkers.load() == 0; });
213  }
214  }
215  DEBUG_THREAD("joining");
216  for (auto& thread : threads) thread->join();
217  DEBUG_THREAD("joined");
218  }
219 
220  void queueFunction(Function* func) {
221  DEBUG_THREAD("queue function");
222  assert(nextFunction < numFunctions); // TODO: if we are given more than we expected, use a slower work queue?
223  list[nextFunction++].store(func);
224  availableFuncs++;
225  }
226 
227  void optimizeGlobally() {
228  PassRunner passRunner(wasm, passOptions);
229  passRunner.addDefaultGlobalOptimizationPasses();
230  passRunner.run();
231  }

235  void optimizeFunction(Function* func) {
236  PassRunner passRunner(wasm, passOptions);
237  addPrePasses(passRunner);
238  passRunner.addDefaultFunctionOptimizationPasses();
239  passRunner.runFunction(func);
240  }

242  static void workerMain(OptimizingIncrementalModuleBuilder* self) {
243  DEBUG_THREAD("workerMain");
244  {
245  std::lock_guard<std::mutex> lock(self->mutex);
246  self->liveWorkers++;
247  self->activeWorkers++;
248  self->condition.notify_all();
249  }
250  for (uint32_t i = 0; i < self->numFunctions; i++) {
251  DEBUG_THREAD("workerMain iteration " << i);
252  if (self->list[i].load() == self->endMarker) {
253  // sleep, this entry isn't ready yet
254  DEBUG_THREAD("workerMain sleep");
255  self->activeWorkers--;
256  {
257  std::unique_lock<std::mutex> lock(self->mutex);
258  if (!self->finishing) { // while waiting for the lock, things may have ended
259  self->condition.wait(lock);
260  }
261  }
262  // continue
263  DEBUG_THREAD("workerMain continue");
264  self->activeWorkers++;
265  i--;
266  continue;
267  }
268  DEBUG_THREAD("workerMain exchange item");
269  auto* func = self->list[i].exchange(nullptr);
270  if (func == nullptr) {
271  DEBUG_THREAD("workerMain sees was already taken");
272  continue; // someone else has taken this one
273  }
274  // we have work to do!
275  DEBUG_THREAD("workerMain work on " << size_t(func));
276  self->availableFuncs--;
277  self->optimizeFunction(func);
278  self->finishedFuncs++;
279  }
280  DEBUG_THREAD("workerMain ready to exit");
281  {
282  std::lock_guard<std::mutex> lock(self->mutex);
283  self->liveWorkers--;
284  self->condition.notify_all();
285  }
286  DEBUG_THREAD("workerMain exiting");
287  }
288 };
