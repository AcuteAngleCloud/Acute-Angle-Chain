35 template<typename SubType, typename ReturnType = void>
36 struct Visitor {
37  // Expression visitors
38  ReturnType visitBlock(Block* curr) {}
39  ReturnType visitIf(If* curr) {}
40  ReturnType visitLoop(Loop* curr) {}
41  ReturnType visitBreak(Break* curr) {}
42  ReturnType visitSwitch(Switch* curr) {}
43  ReturnType visitCall(Call* curr) {}
44  ReturnType visitCallImport(CallImport* curr) {}
45  ReturnType visitCallIndirect(CallIndirect* curr) {}
46  ReturnType visitGetLocal(GetLocal* curr) {}
47  ReturnType visitSetLocal(SetLocal* curr) {}
48  ReturnType visitGetGlobal(GetGlobal* curr) {}
49  ReturnType visitSetGlobal(SetGlobal* curr) {}
50  ReturnType visitLoad(Load* curr) {}
51  ReturnType visitStore(Store* curr) {}
52  ReturnType visitConst(Const* curr) {}
53  ReturnType visitUnary(Unary* curr) {}
54  ReturnType visitBinary(Binary* curr) {}
55  ReturnType visitSelect(Select* curr) {}
56  ReturnType visitDrop(Drop* curr) {}
57  ReturnType visitReturn(Return* curr) {}
58  ReturnType visitHost(Host* curr) {}
59  ReturnType visitNop(Nop* curr) {}
60  ReturnType visitUnreachable(Unreachable* curr) {}
61  // Module-level visitors
62  ReturnType visitFunctionType(FunctionType* curr) {}
63  ReturnType visitImport(Import* curr) {}
64  ReturnType visitExport(Export* curr) {}
65  ReturnType visitGlobal(Global* curr) {}
66  ReturnType visitFunction(Function* curr) {}
67  ReturnType visitTable(Table* curr) {}
68  ReturnType visitMemory(Memory* curr) {}
69  ReturnType visitModule(Module* curr) {}
70 
71  ReturnType visit(Expression* curr) {
72  assert(curr);
73 
74  #define DELEGATE(CLASS_TO_VISIT) \
75  return static_cast<SubType*>(this)-> \
76  visit##CLASS_TO_VISIT(static_cast<CLASS_TO_VISIT*>(curr))
77 
78  switch (curr->_id) {
79  case Expression::Id::BlockId: DELEGATE(Block);
80  case Expression::Id::IfId: DELEGATE(If);
81  case Expression::Id::LoopId: DELEGATE(Loop);
82  case Expression::Id::BreakId: DELEGATE(Break);
83  case Expression::Id::SwitchId: DELEGATE(Switch);
84  case Expression::Id::CallId: DELEGATE(Call);
85  case Expression::Id::CallImportId: DELEGATE(CallImport);
86  case Expression::Id::CallIndirectId: DELEGATE(CallIndirect);
87  case Expression::Id::GetLocalId: DELEGATE(GetLocal);
88  case Expression::Id::SetLocalId: DELEGATE(SetLocal);
89  case Expression::Id::GetGlobalId: DELEGATE(GetGlobal);
90  case Expression::Id::SetGlobalId: DELEGATE(SetGlobal);
91  case Expression::Id::LoadId: DELEGATE(Load);
92  case Expression::Id::StoreId: DELEGATE(Store);
93  case Expression::Id::ConstId: DELEGATE(Const);
94  case Expression::Id::UnaryId: DELEGATE(Unary);
95  case Expression::Id::BinaryId: DELEGATE(Binary);
96  case Expression::Id::SelectId: DELEGATE(Select);
97  case Expression::Id::DropId: DELEGATE(Drop);
98  case Expression::Id::ReturnId: DELEGATE(Return);
99  case Expression::Id::HostId: DELEGATE(Host);
100  case Expression::Id::NopId: DELEGATE(Nop);
101  case Expression::Id::UnreachableId: DELEGATE(Unreachable);
102  case Expression::Id::InvalidId:
103  default: WASM_UNREACHABLE();
104  }
105 
106  #undef DELEGATE
107  }
108 };

144 //
145 // Base class for all WasmWalkers, which can traverse an AST
146 // and provide the option to replace nodes while doing so.
147 //
148 // Subclass and implement the visit*()
149 // calls to run code on different node types.
150 //
151 template<typename SubType, typename VisitorType>
152 struct Walker : public VisitorType {
153  // Useful methods for visitor implementions
154 
155  // Replace the current node. You can call this in your visit*() methods.
156  // Note that the visit*() for the result node is not called for you (i.e.,
157  // just one visit*() method is called by the traversal; if you replace a node,
158  // and you want to process the output, you must do that explicitly).
159  Expression* replaceCurrent(Expression* expression) {
160  return *replacep = expression;
161  }
162 
163  Expression* getCurrent() {
164  return *replacep;
165  }
166 
167  Expression** getCurrentPointer() {
168  return replacep;
169  }
170 
171  // Get the current module
172  Module* getModule() {
173  return currModule;
174  }
175 
176  // Get the current function
177  Function* getFunction() {
178  return currFunction;
179  }
180 
181  // Walk starting
182 
183  void walkGlobal(Global* global) {
184  walk(global->init);
185  static_cast<SubType*>(this)->visitGlobal(global);
186  }
187 
188  void walkFunction(Function* func) {
189  setFunction(func);
190  static_cast<SubType*>(this)->doWalkFunction(func);
191  static_cast<SubType*>(this)->visitFunction(func);
192  setFunction(nullptr);
193  }
194 
195  void walkFunctionInModule(Function* func, Module* module) {
196  setModule(module);
197  setFunction(func);
198  static_cast<SubType*>(this)->doWalkFunction(func);
199  static_cast<SubType*>(this)->visitFunction(func);
200  setFunction(nullptr);
201  setModule(nullptr);
202  }
203 
204  // override this to provide custom functionality
205  void doWalkFunction(Function* func) {
206  walk(func->body);
207  }
208 
209  void walkTable(Table* table) {
210  for (auto& segment : table->segments) {
211  walk(segment.offset);
212  }
213  static_cast<SubType*>(this)->visitTable(table);
214  }
215 
216  void walkMemory(Memory* memory) {
217  for (auto& segment : memory->segments) {
218  walk(segment.offset);
219  }
220  static_cast<SubType*>(this)->visitMemory(memory);
221  }
222 
223  void walkModule(Module* module) {
224  setModule(module);
225  static_cast<SubType*>(this)->doWalkModule(module);
226  static_cast<SubType*>(this)->visitModule(module);
227  setModule(nullptr);
228  }
229 
230  // override this to provide custom functionality
231  void doWalkModule(Module* module) {
232  // Dispatch statically through the SubType.
233  SubType* self = static_cast<SubType*>(this);
234  for (auto& curr : module->functionTypes) {
235  self->visitFunctionType(curr.get());
236  }
237  for (auto& curr : module->imports) {
238  self->visitImport(curr.get());
239  }
240  for (auto& curr : module->exports) {
241  self->visitExport(curr.get());
242  }
243  for (auto& curr : module->globals) {
244  self->walkGlobal(curr.get());
245  }
246  for (auto& curr : module->functions) {
247  self->walkFunction(curr.get());
248  }
249  self->walkTable(&module->table);
250  self->walkMemory(&module->memory);
251  }

253  // Walk implementation. We don't use recursion as ASTs may be highly
254  // nested.

256  // Tasks receive the this pointer and a pointer to the pointer to operate on
257  typedef void (*TaskFunc)(SubType*, Expression**);
258 
259  struct Task {
260  TaskFunc func;
261  Expression** currp;
262  Task(TaskFunc func, Expression** currp) : func(func), currp(currp) {}
263  };
264 
265  void pushTask(TaskFunc func, Expression** currp) {
266  stack.emplace_back(func, currp);
267  }
268  void maybePushTask(TaskFunc func, Expression** currp) {
269  if (*currp) {
270  stack.emplace_back(func, currp);
271  }
272  }
273  Task popTask() {
274  auto ret = stack.back();
275  stack.pop_back();
276  return ret;
277  }
278 
279  void walk(Expression*& root) {
280  assert(stack.size() == 0);
281  pushTask(SubType::scan, &root);
282  while (stack.size() > 0) {
283  auto task = popTask();
284  replacep = task.currp;
285  assert(*task.currp);
286  task.func(static_cast<SubType*>(this), task.currp);
287  }
288  }

334 // Walks in post-order, i.e., children first. When there isn't an obvious
335 // order to operands, we follow them in order of execution.
336 
337 template<typename SubType, typename VisitorType = Visitor<SubType>>
338 struct PostWalker : public Walker<SubType, VisitorType> {
339 
340  static void scan(SubType* self, Expression** currp) {
341 
342  Expression* curr = *currp;
343  switch (curr->_id) {
344  case Expression::Id::InvalidId: abort();
345  case Expression::Id::BlockId: {
346  self->pushTask(SubType::doVisitBlock, currp);
347  auto& list = curr->cast<Block>()->list;
348  for (int i = int(list.size()) - 1; i >= 0; i--) {
349  self->pushTask(SubType::scan, &list[i]);
350  }
351  break;
352  }
353  case Expression::Id::IfId: {
354  self->pushTask(SubType::doVisitIf, currp);
355  self->maybePushTask(SubType::scan, &curr->cast<If>()->ifFalse);
356  self->pushTask(SubType::scan, &curr->cast<If>()->ifTrue);
357  self->pushTask(SubType::scan, &curr->cast<If>()->condition);
358  break;
359  }
360  case Expression::Id::LoopId: {
361  self->pushTask(SubType::doVisitLoop, currp);
362  self->pushTask(SubType::scan, &curr->cast<Loop>()->body);
363  break;
364  }
365  case Expression::Id::BreakId: {
366  self->pushTask(SubType::doVisitBreak, currp);
367  self->maybePushTask(SubType::scan, &curr->cast<Break>()->condition);
368  self->maybePushTask(SubType::scan, &curr->cast<Break>()->value);
369  break;
370  }
371  case Expression::Id::SwitchId: {
372  self->pushTask(SubType::doVisitSwitch, currp);
373  self->pushTask(SubType::scan, &curr->cast<Switch>()->condition);
374  self->maybePushTask(SubType::scan, &curr->cast<Switch>()->value);
375  break;
376  }
377  case Expression::Id::CallId: {
378  self->pushTask(SubType::doVisitCall, currp);
379  auto& list = curr->cast<Call>()->operands;
380  for (int i = int(list.size()) - 1; i >= 0; i--) {
381  self->pushTask(SubType::scan, &list[i]);
382  }
383  break;
384  }
385  case Expression::Id::CallImportId: {
386  self->pushTask(SubType::doVisitCallImport, currp);
387  auto& list = curr->cast<CallImport>()->operands;
388  for (int i = int(list.size()) - 1; i >= 0; i--) {
389  self->pushTask(SubType::scan, &list[i]);
390  }
391  break;
392  }
393  case Expression::Id::CallIndirectId: {
394  self->pushTask(SubType::doVisitCallIndirect, currp);
395  auto& list = curr->cast<CallIndirect>()->operands;
396  self->pushTask(SubType::scan, &curr->cast<CallIndirect>()->target);
397  for (int i = int(list.size()) - 1; i >= 0; i--) {
398  self->pushTask(SubType::scan, &list[i]);
399  }
400  break;
401  }
402  case Expression::Id::GetLocalId: {
403  self->pushTask(SubType::doVisitGetLocal, currp); // TODO: optimize leaves with a direct call?
404  break;
405  }
406  case Expression::Id::SetLocalId: {
407  self->pushTask(SubType::doVisitSetLocal, currp);
408  self->pushTask(SubType::scan, &curr->cast<SetLocal>()->value);
409  break;
410  }
411  case Expression::Id::GetGlobalId: {
412  self->pushTask(SubType::doVisitGetGlobal, currp);
413  break;
414  }
415  case Expression::Id::SetGlobalId: {
416  self->pushTask(SubType::doVisitSetGlobal, currp);
417  self->pushTask(SubType::scan, &curr->cast<SetGlobal>()->value);
418  break;
419  }
420  case Expression::Id::LoadId: {
421  self->pushTask(SubType::doVisitLoad, currp);
422  self->pushTask(SubType::scan, &curr->cast<Load>()->ptr);
423  break;
424  }
425  case Expression::Id::StoreId: {
426  self->pushTask(SubType::doVisitStore, currp);
427  self->pushTask(SubType::scan, &curr->cast<Store>()->value);
428  self->pushTask(SubType::scan, &curr->cast<Store>()->ptr);
429  break;
430  }
431  case Expression::Id::ConstId: {
432  self->pushTask(SubType::doVisitConst, currp);
433  break;
434  }
435  case Expression::Id::UnaryId: {
436  self->pushTask(SubType::doVisitUnary, currp);
437  self->pushTask(SubType::scan, &curr->cast<Unary>()->value);
438  break;
439  }
440  case Expression::Id::BinaryId: {
441  self->pushTask(SubType::doVisitBinary, currp);
442  self->pushTask(SubType::scan, &curr->cast<Binary>()->right);
443  self->pushTask(SubType::scan, &curr->cast<Binary>()->left);
444  break;
445  }
446  case Expression::Id::SelectId: {
447  self->pushTask(SubType::doVisitSelect, currp);
448  self->pushTask(SubType::scan, &curr->cast<Select>()->condition);
449  self->pushTask(SubType::scan, &curr->cast<Select>()->ifFalse);
450  self->pushTask(SubType::scan, &curr->cast<Select>()->ifTrue);
451  break;
452  }
453  case Expression::Id::DropId: {
454  self->pushTask(SubType::doVisitDrop, currp);
455  self->pushTask(SubType::scan, &curr->cast<Drop>()->value);
456  break;
457  }
458  case Expression::Id::ReturnId: {
459  self->pushTask(SubType::doVisitReturn, currp);
460  self->maybePushTask(SubType::scan, &curr->cast<Return>()->value);
461  break;
462  }
463  case Expression::Id::HostId: {
464  self->pushTask(SubType::doVisitHost, currp);
465  auto& list = curr->cast<Host>()->operands;
466  for (int i = int(list.size()) - 1; i >= 0; i--) {
467  self->pushTask(SubType::scan, &list[i]);
468  }
469  break;
470  }
471  case Expression::Id::NopId: {
472  self->pushTask(SubType::doVisitNop, currp);
473  break;
474  }
475  case Expression::Id::UnreachableId: {
476  self->pushTask(SubType::doVisitUnreachable, currp);
477  break;
478  }
479  default: WASM_UNREACHABLE();
480  }
481  }
482 };

484 // Traversal with a control-flow stack.

486 template<typename SubType, typename VisitorType = Visitor<SubType>>
487 struct ControlFlowWalker : public PostWalker<SubType, VisitorType> {
488  ControlFlowWalker() {}
489 
490  std::vector<Expression*> controlFlowStack; // contains blocks, loops, and ifs
491 
492  // Uses the control flow stack to find the target of a break to a name
493  Expression* findBreakTarget(Name name) {
494  assert(!controlFlowStack.empty());
495  Index i = controlFlowStack.size() - 1;
496  while (1) {
497  auto* curr = controlFlowStack[i];
498  if (Block* block = curr->template dynCast<Block>()) {
499  if (name == block->name) return curr;
500  } else if (Loop* loop = curr->template dynCast<Loop>()) {
501  if (name == loop->name) return curr;
502  } else {
503  // an if, ignorable
504  assert(curr->template is<If>());
505  }
506  if (i == 0) return nullptr;
507  i--;
508  }
509  }
510 
511  static void doPreVisitControlFlow(SubType* self, Expression** currp) {
512  self->controlFlowStack.push_back(*currp);
513  }
514 
515  static void doPostVisitControlFlow(SubType* self, Expression** currp) {
516  assert(self->controlFlowStack.back() == *currp);
517  self->controlFlowStack.pop_back();
518  }
519 
520  static void scan(SubType* self, Expression** currp) {
521  auto* curr = *currp;
522 
523  switch (curr->_id) {
524  case Expression::Id::BlockId:
525  case Expression::Id::IfId:
526  case Expression::Id::LoopId: {
527  self->pushTask(SubType::doPostVisitControlFlow, currp);
528  break;
529  }
530  default: {}
531  }
532 
533  PostWalker<SubType, VisitorType>::scan(self, currp);
534 
535  switch (curr->_id) {
536  case Expression::Id::BlockId:
537  case Expression::Id::IfId:
538  case Expression::Id::LoopId: {
539  self->pushTask(SubType::doPreVisitControlFlow, currp);
540  break;
541  }
542  default: {}
543  }
544  }
545 };

597 // Traversal in the order of execution. This is quick and simple, but
598 // does not provide the same comprehensive information that a full
599 // conversion to basic blocks would. What it does give is a quick
600 // way to view straightline execution traces, i.e., that have no
601 // branching. This can let optimizations get most of what they
602 // want without the cost of creating another AST.
603 //
604 // When execution is no longer linear, this notifies via a call
605 // to noteNonLinear().
606 
607 template<typename SubType, typename VisitorType = Visitor<SubType>>
608 struct LinearExecutionWalker : public PostWalker<SubType, VisitorType> {
609  LinearExecutionWalker() {}
610 
611  // subclasses should implement this
612  void noteNonLinear(Expression* curr) { abort(); }
613 
614  static void doNoteNonLinear(SubType* self, Expression** currp) {
615  self->noteNonLinear(*currp);
616  }
617 
618  static void scan(SubType* self, Expression** currp) {
619 
620  Expression* curr = *currp;
621 
622  switch (curr->_id) {
623  case Expression::Id::InvalidId: abort();
624  case Expression::Id::BlockId: {
625  self->pushTask(SubType::doVisitBlock, currp);
626  if (curr->cast<Block>()->name.is()) {
627  self->pushTask(SubType::doNoteNonLinear, currp);
628  }
629  auto& list = curr->cast<Block>()->list;
630  for (int i = int(list.size()) - 1; i >= 0; i--) {
631  self->pushTask(SubType::scan, &list[i]);
632  }
633  break;
634  }
635  case Expression::Id::IfId: {
636  self->pushTask(SubType::doVisitIf, currp);
637  self->pushTask(SubType::doNoteNonLinear, currp);
638  self->maybePushTask(SubType::scan, &curr->cast<If>()->ifFalse);
639  self->pushTask(SubType::doNoteNonLinear, currp);
640  self->pushTask(SubType::scan, &curr->cast<If>()->ifTrue);
641  self->pushTask(SubType::doNoteNonLinear, currp);
642  self->pushTask(SubType::scan, &curr->cast<If>()->condition);
643  break;
644  }
645  case Expression::Id::LoopId: {
646  self->pushTask(SubType::doVisitLoop, currp);
647  self->pushTask(SubType::scan, &curr->cast<Loop>()->body);
648  self->pushTask(SubType::doNoteNonLinear, currp);
649  break;
650  }
651  case Expression::Id::BreakId: {
652  self->pushTask(SubType::doVisitBreak, currp);
653  self->pushTask(SubType::doNoteNonLinear, currp);
654  self->maybePushTask(SubType::scan, &curr->cast<Break>()->condition);
655  self->maybePushTask(SubType::scan, &curr->cast<Break>()->value);
656  break;
657  }
658  case Expression::Id::SwitchId: {
659  self->pushTask(SubType::doVisitSwitch, currp);
660  self->pushTask(SubType::doNoteNonLinear, currp);
661  self->maybePushTask(SubType::scan, &curr->cast<Switch>()->value);
662  self->pushTask(SubType::scan, &curr->cast<Switch>()->condition);
663  break;
664  }
665  case Expression::Id::ReturnId: {
666  self->pushTask(SubType::doVisitReturn, currp);
667  self->pushTask(SubType::doNoteNonLinear, currp);
668  self->maybePushTask(SubType::scan, &curr->cast<Return>()->value);
669  break;
670  }
671  case Expression::Id::UnreachableId: {
672  self->pushTask(SubType::doVisitUnreachable, currp);
673  self->pushTask(SubType::doNoteNonLinear, currp);
674  break;
675  }
676  default: {
677  // other node types do not have control flow, use regular post-order
678  PostWalker<SubType, VisitorType>::scan(self, currp);
679  }
680  }
681  }
682 };
