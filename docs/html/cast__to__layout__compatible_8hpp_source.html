18 template <class T, class U>
19 constexpr void static_assert_layout_compatible() noexcept {
20  static_assert(
21  std::alignment_of<T>::value == std::alignment_of<U>::value,
22  "Alignment check failed, probably your structure has user-defined alignment for the whole structure or for some of the fields."
23  );
24  static_assert(sizeof(T) == sizeof(U), "Size check failed, probably your structure has bitfields or user-defined alignment.");
25 }
