35 struct ShellExternalInterface final : ModuleInstance::ExternalInterface {
36  // The underlying memory can be accessed through unaligned pointers which
37  // isn't well-behaved in C++. WebAssembly nonetheless expects it to behave
38  // properly. Avoid emitting unaligned load/store by checking for alignment
39  // explicitly, and performing memcpy if unaligned.
40  //
41  // The allocated memory tries to have the same alignment as the memory being
42  // simulated.
43  class Memory {
44  // Use char because it doesn't run afoul of aliasing rules.
45  std::vector<char> memory;
46  template <typename T>
47  static bool aligned(const char* address) {
48  static_assert(!(sizeof(T) & (sizeof(T) - 1)), "must be a power of 2");
49  return 0 == (reinterpret_cast<uintptr_t>(address) & (sizeof(T) - 1));
50  }
51  Memory(Memory&) = delete;
52  Memory& operator=(const Memory&) = delete;
53 
54  public:
55  Memory() {}
56  void resize(size_t newSize) {
57  // Ensure the smallest allocation is large enough that most allocators
58  // will provide page-aligned storage. This hopefully allows the
59  // interpreter's memory to be as aligned as the memory being simulated,
60  // ensuring that the performance doesn't needlessly degrade.
61  //
62  // The code is optimistic this will work until WG21's p0035r0 happens.
63  const size_t minSize = 1 << 12;
64  size_t oldSize = memory.size();
65  memory.resize(std::max(minSize, newSize));
66  if (newSize < oldSize && newSize < minSize) {
67  std::memset(&memory[newSize], 0, minSize - newSize);
68  }
69  }
70  template <typename T>
71  void set(size_t address, T value) {
72  if (aligned<T>(&memory[address])) {
73  *reinterpret_cast<T*>(&memory[address]) = value;
74  } else {
75  std::memcpy(&memory[address], &value, sizeof(T));
76  }
77  }
78  template <typename T>
79  T get(size_t address) {
80  if (aligned<T>(&memory[address])) {
81  return *reinterpret_cast<T*>(&memory[address]);
82  } else {
83  T loaded;
84  std::memcpy(&loaded, &memory[address], sizeof(T));
85  return loaded;
86  }
87  }
88  } memory;

94  void init(Module& wasm, ModuleInstance& instance) override {
95  memory.resize(wasm.memory.initial * wasm::Memory::kPageSize);
96  // apply memory segments
97  for (auto& segment : wasm.memory.segments) {
98  Address offset = ConstantExpressionRunner<TrivialGlobalManager>(instance.globals).visit(segment.offset).value.geti32();
99  assert(offset + segment.data.size() <= wasm.memory.initial * wasm::Memory::kPageSize);
100  for (size_t i = 0; i != segment.data.size(); ++i) {
101  memory.set(offset + i, segment.data[i]);
102  }
103  }
104 
105  table.resize(wasm.table.initial);
106  for (auto& segment : wasm.table.segments) {
107  Address offset = ConstantExpressionRunner<TrivialGlobalManager>(instance.globals).visit(segment.offset).value.geti32();
108  assert(offset + segment.data.size() <= wasm.table.initial);
109  for (size_t i = 0; i != segment.data.size(); ++i) {
110  table[offset + i] = segment.data[i];
111  }
112  }
113  }

115  void importGlobals(std::map<Name, Literal>& globals, Module& wasm) override {
116  // add spectest globals
117  for (auto& import : wasm.imports) {
118  if (import->kind == ExternalKind::Global && import->module == SPECTEST && import->base == GLOBAL) {
119  switch (import->globalType) {
120  case i32: globals[import->name] = Literal(int32_t(666)); break;
121  case i64: globals[import->name] = Literal(int64_t(666)); break;
122  case f32: globals[import->name] = Literal(float(666.6)); break;
123  case f64: globals[import->name] = Literal(double(666.6)); break;
124  default: WASM_UNREACHABLE();
125  }
126  } else if (import->kind == ExternalKind::Memory && import->module == SPECTEST && import->base == MEMORY) {
127  // imported memory has initial 1 and max 2
128  wasm.memory.initial = 1;
129  wasm.memory.max = 2;
130  }
131  }

149  Literal callTable(Index index, LiteralList& arguments, WasmType result, ModuleInstance& instance) override {
150  if (index >= table.size()) trap("callTable overflow");
151  auto* func = instance.wasm.getFunctionOrNull(table[index]);
152  if (!func) trap("uninitialized table element");
153  if (func->params.size() != arguments.size()) trap("callIndirect: bad # of arguments");
154  for (size_t i = 0; i < func->params.size(); i++) {
155  if (func->params[i] != arguments[i].type) {
156  trap("callIndirect: bad argument type");
157  }
158  }
159  if (func->result != result) {
160  trap("callIndirect: bad result type");
161  }
162  return instance.callFunctionInternal(func->name, arguments);
163  }
