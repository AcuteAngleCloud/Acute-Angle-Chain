69 template <typename T, typename U>
70 inline static T RotateLeft(T val, U count) {
71  T mask = sizeof(T) * CHAR_BIT - 1;
72  count &= mask;
73  return (val << count) | (val >> (-count & mask));
74 }
75 template <typename T, typename U>
76 inline static T RotateRight(T val, U count) {
77  T mask = sizeof(T) * CHAR_BIT - 1;
78  count &= mask;
79  return (val >> count) | (val << (-count & mask));
80 }
