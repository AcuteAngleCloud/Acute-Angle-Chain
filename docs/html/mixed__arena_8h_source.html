28 //
29 // Arena allocation for mixed-type data.
30 //
31 // Arena-style bump allocation is important for two reasons: First, so that
32 // allocation is quick, and second, so that allocated items are close together,
33 // which is cache-friendy. Arena allocation is also useful for a minor third
34 // reason which is to make freeing all the items in an arena very quick.
35 //
36 // Each WebAssembly Module has an arena allocator, which should be used
37 // for all of its AST nodes and so forth. When the Module is destroyed, the
38 // entire arena is cleaned up.
39 //
40 // When allocating an object in an arena, the object's proper constructor
41 // is called. Note that destructors are not called, because to make the
42 // arena simple and fast we do not track internal allocations inside it
43 // (and we can also avoid the need for virtual destructors).
44 //
45 // In general, optimization passes avoid allocation as much as possible.
46 // Many passes only remove or modify nodes anyhow, others can often
47 // reuse nodes that are being optimized out. This keeps things
48 // cache-friendly, and also makes the operations trivially thread-safe.
49 // In the rare case that a pass does need to allocate, and it is a
50 // parallel pass (so multiple threads might access the allocator),
51 // the MixedArena instance will notice if it is on a different thread
52 // than that arena's original thread, and will perform the allocation
53 // in a side arena for that other thread. This is done in a transparent
54 // way to the outside; as a result, it is always safe to allocate using
55 // a MixedArena, no matter which thread you are on. Allocations will
56 // of course be fastest on the original thread for the arena.
57 //
58 
59 struct MixedArena {
60  // fast bump allocation
61  std::vector<char*> chunks;
62  size_t chunkSize = 32768;
63  size_t index; // in last chunk
64 
65  std::thread::id threadId;
66 
67  // multithreaded allocation - each arena is valid on a specific thread.
68  // if we are on the wrong thread, we atomically look in the linked
69  // list of next, adding an allocator if necessary
70  std::atomic<MixedArena*> next;
71 
72  MixedArena() {
73  threadId = std::this_thread::get_id();
74  next.store(nullptr);
75  }
76 
77  void* allocSpace(size_t size) {
78  // the bump allocator data should not be modified by multiple threads at once.
79  auto myId = std::this_thread::get_id();
80  if (myId != threadId) {
81  MixedArena* curr = this;
82  MixedArena* allocated = nullptr;
83  while (myId != curr->threadId) {
84  auto seen = curr->next.load();
85  if (seen) {
86  curr = seen;
87  continue;
88  }
89  // there is a nullptr for next, so we may be able to place a new
90  // allocator for us there. but carefully, as others may do so as
91  // well. we may waste a few allocations here, but it doesn't matter
92  // as this can only happen as the chain is built up, i.e.,
93  // O(# of cores) per allocator, and our allocatrs are long-lived.
94  if (!allocated) {
95  allocated = new MixedArena(); // has our thread id
96  }
97  if (curr->next.compare_exchange_weak(seen, allocated)) {
98  // we replaced it, so we are the next in the chain
99  // we can forget about allocated, it is owned by the chain now
100  allocated = nullptr;
101  break;
102  }
103  // otherwise, the cmpxchg updated seen, and we continue to loop
104  curr = seen;
105  }
106  if (allocated) delete allocated;
107  return curr->allocSpace(size);
108  }
109  size = (size + 7) & (-8); // same alignment as malloc TODO optimize?
110  bool mustAllocate = false;
111  while (chunkSize <= size) {
112  chunkSize *= 2;
113  mustAllocate = true;
114  }
115  if (chunks.size() == 0 || index + size >= chunkSize || mustAllocate) {
116  chunks.push_back(new char[chunkSize]);
117  index = 0;
118  }
119  auto* ret = chunks.back() + index;
120  index += size;
121  return static_cast<void*>(ret);
122  }
123 
124  template<class T>
125  T* alloc() {
126  auto* ret = static_cast<T*>(allocSpace(sizeof(T)));
127  new (ret) T(*this); // allocated objects receive the allocator, so they can allocate more later if necessary
128  return ret;
129  }
130 
131  void clear() {
132  for (char* chunk : chunks) {
133  delete[] chunk;
134  }
135  chunks.clear();
136  }
137 
138  ~MixedArena() {
139  clear();
140  if (next.load()) delete next.load();
141  }
142 };

150 template <typename SubType, typename T>
151 class ArenaVectorBase {
152 protected:
153  T* data = nullptr;
154  size_t usedElements = 0,
155  allocatedElements = 0;
156 
157  void reallocate(size_t size) {
158  T* old = data;
159  static_cast<SubType*>(this)->allocate(size);
160  for (size_t i = 0; i < usedElements; i++) {
161  data[i] = old[i];
162  }
163  }
164 
165 public:
166  T& operator[](size_t index) const {
167  assert(index < usedElements);
168  return data[index];
169  }
170 
171  size_t size() const {
172  return usedElements;
173  }
174 
175  void resize(size_t size) {
176  if (size > allocatedElements) {
177  reallocate(size);
178  }
179  // construct new elements
180  for (size_t i = usedElements; i < size; i++) {
181  new (data + i) T();
182  }
183  usedElements = size;
184  }
185 
186  T& back() const {
187  assert(usedElements > 0);
188  return data[usedElements - 1];
189  }
190 
191  T& pop_back() {
192  assert(usedElements > 0);
193  usedElements--;
194  return data[usedElements];
195  }
196 
197  void push_back(T item) {
198  if (usedElements == allocatedElements) {
199  reallocate((allocatedElements + 1) * 2); // TODO: optimize
200  }
201  data[usedElements] = item;
202  usedElements++;
203  }
204 
205  void clear() {
206  usedElements = 0;
207  }
208 
209  void reserve(size_t size) {
210  if (size > allocatedElements) {
211  reallocate(size);
212  }
213  }

216  void set(const ListType& list) {
217  size_t size = list.size();
218  if (allocatedElements < size) {
219  static_cast<SubType*>(this)->allocate(size);
220  }
