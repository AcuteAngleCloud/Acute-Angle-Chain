39 template<typename T, typename MiniT>
40 struct LEB {
41  static_assert(sizeof(MiniT) == 1, "MiniT must be a byte");
42 
43  T value;
44 
45  LEB() {}
46  LEB(T value) : value(value) {}
47 
48  bool hasMore(T temp, MiniT byte) {
49  // for signed, we must ensure the last bit has the right sign, as it will zero extend
50  return std::is_signed<T>::value ? (temp != 0 && temp != -1) || (value >= 0 && (byte & 64)) || (value < 0 && !(byte & 64)) : (temp != 0);
51  }
52 
53  void write(std::vector<uint8_t>* out) {
54  T temp = value;
55  bool more;
56  do {
57  uint8_t byte = temp & 127;
58  temp >>= 7;
59  more = hasMore(temp, byte);
60  if (more) {
61  byte = byte | 128;
62  }
63  out->push_back(byte);
64  } while (more);
65  }
66 
67  void writeAt(std::vector<uint8_t>* out, size_t at, size_t minimum = 0) {
68  T temp = value;
69  size_t offset = 0;
70  bool more;
71  do {
72  uint8_t byte = temp & 127;
73  temp >>= 7;
74  more = hasMore(temp, byte) || offset + 1 < minimum;
75  if (more) {
76  byte = byte | 128;
77  }
78  (*out)[at + offset] = byte;
79  offset++;
80  } while (more);
81  }
82 
83  void read(std::function<MiniT()> get) {
84  value = 0;
85  T shift = 0;
86  MiniT byte;
87  while (1) {
88  byte = get();
89  bool last = !(byte & 128);
90  T payload = byte & 127;
91  typedef typename std::make_unsigned<T>::type mask_type;
92  auto shift_mask = 0 == shift
93  ? ~mask_type(0)
94  : ((mask_type(1) << (sizeof(T) * 8 - shift)) - 1u);
95  T significant_payload = payload & shift_mask;
96  if (significant_payload != payload) {
97  if (!(std::is_signed<T>::value && last)) {
98  throw ParseException("LEB dropped bits only valid for signed LEB");
99  }
100  }
101  value |= significant_payload << shift;
102  if (last) break;
103  shift += 7;
104  if (size_t(shift) >= sizeof(T) * 8) {
105  throw ParseException("LEB overflow");
106  }
107  }
108  // If signed LEB, then we might need to sign-extend. (compile should
109  // optimize this out if not needed).
110  if (std::is_signed<T>::value) {
111  shift += 7;
112  if ((byte & 64) && size_t(shift) < 8 * sizeof(T)) {
113  size_t sext_bits = 8 * sizeof(T) - size_t(shift);
114  value <<= sext_bits;
115  value >>= sext_bits;
116  if (value >= 0) {
117  throw ParseException(" LEBsign-extend should produce a negative value");
118  }
119  }
120  }
121  }
122 };

251  if (debug) std::cerr << "backpatchInt16: " << x << " (at " << i << ")" << std::endl;

263  if (debug) std::cerr << "backpatchU32LEB: " << x.value << " (at " << i << ")" << std::endl;

521 inline S32LEB binaryWasmType(WasmType type) {
522  int ret;
523  switch (type) {
524  // None only used for block signatures. TODO: Separate out?
525  case none: ret = BinaryConsts::EncodedType::Empty; break;
526  case i32: ret = BinaryConsts::EncodedType::i32; break;
527  case i64: ret = BinaryConsts::EncodedType::i64; break;
528  case f32: ret = BinaryConsts::EncodedType::f32; break;
529  case f64: ret = BinaryConsts::EncodedType::f64; break;
530  default: abort();
531  }
532  return S32LEB(ret);
533 }
