163  const DigestType& append(const DigestType& digest) {
164  bool partial = false;
165  auto max_depth = detail::calcluate_max_depth(_node_count + 1);
166  auto current_depth = max_depth - 1;
167  auto index = _node_count;
168  auto top = digest;
169  auto active_iter = _active_nodes.begin();
170  auto updated_active_nodes = vector<DigestType>();
171  updated_active_nodes.reserve(max_depth);
172 
173  while (current_depth > 0) {
174  if (!(index & 0x1)) {
175  // we are collapsing from a "left" value and an implied "right" creating a partial node
176 
177  // we only need to append this node if it is fully-realized and by definition
178  // if we have encountered a partial node during collapse this cannot be
179  // fully-realized
180  if (!partial) {
181  updated_active_nodes.emplace_back(top);
182  }
183 
184  // calculate the partially realized node value by implying the "right" value is identical
185  // to the "left" value
186  top = DigestType::hash(make_canonical_pair(top, top));
187  partial = true;
188  } else {
189  // we are collapsing from a "right" value and an fully-realized "left"
190 
191  // pull a "left" value from the previous active nodes
192  const auto& left_value = *active_iter;
193  ++active_iter;
194 
195  // if the "right" value is a partial node we will need to copy the "left" as future appends still need it
196  // otherwise, it can be dropped from the set of active nodes as we are collapsing a fully-realized node
197  if (partial) {
198  updated_active_nodes.emplace_back(left_value);
199  }
200 
201  // calculate the node
202  top = DigestType::hash(make_canonical_pair(left_value, top));
203  }
204 
205  // move up a level in the tree
206  current_depth--;
207  index = index >> 1;
208  }
209 
210  // append the top of the collapsed tree (aka the root of the merkle)
211  updated_active_nodes.emplace_back(top);
212 
213  // store the new active_nodes
214  detail::move_nodes(_active_nodes, std::move(updated_active_nodes));
215 
216  // update the node count
217  _node_count++;
218 
219  return _active_nodes.back();
220 
221  }
