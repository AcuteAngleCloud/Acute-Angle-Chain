79 inline Expression* parseConst(cashew::IString s, WasmType type, MixedArena& allocator) {
80  const char *str = s.str;
81  auto ret = allocator.alloc<Const>();
82  ret->type = type;
83  if (isWasmTypeFloat(type)) {
84  if (s == _INFINITY) {
85  switch (type) {
86  case f32: ret->value = Literal(std::numeric_limits<float>::infinity()); break;
87  case f64: ret->value = Literal(std::numeric_limits<double>::infinity()); break;
88  default: return nullptr;
89  }
90  //std::cerr << "make constant " << str << " ==> " << ret->value << '\n';
91  return ret;
92  }
93  if (s == NEG_INFINITY) {
94  switch (type) {
95  case f32: ret->value = Literal(-std::numeric_limits<float>::infinity()); break;
96  case f64: ret->value = Literal(-std::numeric_limits<double>::infinity()); break;
97  default: return nullptr;
98  }
99  //std::cerr << "make constant " << str << " ==> " << ret->value << '\n';
100  return ret;
101  }
102  if (s == _NAN) {
103  switch (type) {
104  case f32: ret->value = Literal(float(std::nan(""))); break;
105  case f64: ret->value = Literal(double(std::nan(""))); break;
106  default: return nullptr;
107  }
108  //std::cerr << "make constant " << str << " ==> " << ret->value << '\n';
109  return ret;
110  }
111  bool negative = str[0] == '-';
112  const char *positive = negative ? str + 1 : str;
113  if (!negative) {
114  if (positive[0] == '+') {
115  positive++;
116  }
117  }
118  if (positive[0] == 'n' && positive[1] == 'a' && positive[2] == 'n') {
119  const char * modifier = positive[3] == ':' ? positive + 4 : nullptr;
120  if (!(modifier ? positive[4] == '0' && positive[5] == 'x' : 1)) {
121  throw ParseException("bad nan input");
122  }
123  switch (type) {
124  case f32: {
125  uint32_t pattern;
126  if (modifier) {
127  std::istringstream istr(modifier);
128  istr >> std::hex >> pattern;
129  pattern |= 0x7f800000U;
130  } else {
131  pattern = 0x7fc00000U;
132  }
133  if (negative) pattern |= 0x80000000U;
134  if (!std::isnan(bit_cast<float>(pattern))) pattern |= 1U;
135  ret->value = Literal(pattern).castToF32();
136  break;
137  }
138  case f64: {
139  uint64_t pattern;
140  if (modifier) {
141  std::istringstream istr(modifier);
142  istr >> std::hex >> pattern;
143  pattern |= 0x7ff0000000000000ULL;
144  } else {
145  pattern = 0x7ff8000000000000UL;
146  }
147  if (negative) pattern |= 0x8000000000000000ULL;
148  if (!std::isnan(bit_cast<double>(pattern))) pattern |= 1ULL;
149  ret->value = Literal(pattern).castToF64();
150  break;
151  }
152  default: return nullptr;
153  }
154  //std::cerr << "make constant " << str << " ==> " << ret->value << '\n';
155  return ret;
156  }
157  if (s == NEG_NAN) {
158  switch (type) {
159  case f32: ret->value = Literal(float(-std::nan(""))); break;
160  case f64: ret->value = Literal(double(-std::nan(""))); break;
161  default: return nullptr;
162  }
163  //std::cerr << "make constant " << str << " ==> " << ret->value << '\n';
164  return ret;
165  }
166  }
167  switch (type) {
168  case i32: {
169  if ((str[0] == '0' && str[1] == 'x') || (str[0] == '-' && str[1] == '0' && str[2] == 'x')) {
170  bool negative = str[0] == '-';
171  if (negative) str++;
172  std::istringstream istr(str);
173  uint32_t temp;
174  istr >> std::hex >> temp;
175  ret->value = Literal(negative ? -temp : temp);
176  } else {
177  std::istringstream istr(str[0] == '-' ? str + 1 : str);
178  uint32_t temp;
179  istr >> temp;
180  ret->value = Literal(str[0] == '-' ? -temp : temp);
181  }
182  break;
183  }
184  case i64: {
185  if ((str[0] == '0' && str[1] == 'x') || (str[0] == '-' && str[1] == '0' && str[2] == 'x')) {
186  bool negative = str[0] == '-';
187  if (negative) str++;
188  std::istringstream istr(str);
189  uint64_t temp;
190  istr >> std::hex >> temp;
191  ret->value = Literal(negative ? -temp : temp);
192  } else {
193  std::istringstream istr(str[0] == '-' ? str + 1 : str);
194  uint64_t temp;
195  istr >> temp;
196  ret->value = Literal(str[0] == '-' ? -temp : temp);
197  }
198  break;
199  }
200  case f32: {
201  char *end;
202  ret->value = Literal(strtof(str, &end));
203  break;
204  }
205  case f64: {
206  char *end;
207  ret->value = Literal(strtod(str, &end));
208  break;
209  }
210  default: return nullptr;
211  }
212  if (ret->value.type != type) {
213  throw ParseException("parsed type does not match expected type");
214  }
215  //std::cerr << "make constant " << str << " ==> " << ret->value << '\n';
216  return ret;
217 }

219 // Helper for parsers that may not have unique label names. This transforms
220 // the names into unique ones, as required by Binaryen IR.
221 struct UniqueNameMapper {
222  std::vector<Name> labelStack;
223  std::map<Name, std::vector<Name>> labelMappings; // name in source => stack of uniquified names
224  std::map<Name, Name> reverseLabelMapping; // uniquified name => name in source
225 
226  Index otherIndex = 0;
227 
228  Name getPrefixedName(Name prefix) {
229  if (reverseLabelMapping.find(prefix) == reverseLabelMapping.end()) return prefix;
230  // make sure to return a unique name not already on the stack
231  while (1) {
232  Name ret = Name(prefix.str + std::to_string(otherIndex++));
233  if (reverseLabelMapping.find(ret) == reverseLabelMapping.end()) return ret;
234  }
235  }
236 
237  // receives a source name. generates a unique name, pushes it, and returns it
238  Name pushLabelName(Name sName) {
239  Name name = getPrefixedName(sName);
240  labelStack.push_back(name);
241  labelMappings[sName].push_back(name);
242  reverseLabelMapping[name] = sName;
243  return name;
244  }
245 
246  void popLabelName(Name name) {
247  assert(labelStack.back() == name);
248  labelStack.pop_back();
249  labelMappings[reverseLabelMapping[name]].pop_back();
250  }
251 
252  Name sourceToUnique(Name sName) {
253  if (labelMappings.find(sName) == labelMappings.end()) {
254  throw ParseException("bad label in sourceToUnique");
255  }
256  if (labelMappings[sName].empty()) {
257  throw ParseException("use of popped label in sourceToUnique");
258  }
259  return labelMappings[sName].back();
260  }
261 
262  Name uniqueToSource(Name name) {
263  if (reverseLabelMapping.find(name) == reverseLabelMapping.end()) {
264  throw ParseException("label mismatch in uniqueToSource");
265  }
266  return reverseLabelMapping[name];
267  }
268 
269  void clear() {
270  labelStack.clear();
271  labelMappings.clear();
272  reverseLabelMapping.clear();
273  }

275  // Given an expression, ensures all names are unique
276  static void uniquify(Expression* curr) {
277  struct Walker : public ControlFlowWalker<Walker, Visitor<Walker>> {
278  UniqueNameMapper mapper;
279 
280  static void doPreVisitControlFlow(Walker* self, Expression** currp) {
281  auto* curr = *currp;
282  if (auto* block = curr->dynCast<Block>()) {
283  if (block->name.is()) block->name = self->mapper.pushLabelName(block->name);
284  } else if (auto* loop = curr->dynCast<Loop>()) {
285  if (loop->name.is()) loop->name = self->mapper.pushLabelName(loop->name);
286  }
287  }
288  static void doPostVisitControlFlow(Walker* self, Expression** currp) {
289  auto* curr = *currp;
290  if (auto* block = curr->dynCast<Block>()) {
291  if (block->name.is()) self->mapper.popLabelName(block->name);
292  } else if (auto* loop = curr->dynCast<Loop>()) {
293  if (loop->name.is()) self->mapper.popLabelName(loop->name);
294  }
295  }
296 
297  void visitBreak(Break *curr) {
298  curr->name = mapper.sourceToUnique(curr->name);
299  }
300  void visitSwitch(Switch *curr) {
301  for (auto& target : curr->targets) {
302  target = mapper.sourceToUnique(target);
303  }
304  curr->default_ = mapper.sourceToUnique(curr->default_);
305  }
306  };
307 
308  Walker walker;
309  walker.walk(curr);
310  }
311 };
