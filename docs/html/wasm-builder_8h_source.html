45  Function* makeFunction(Name name,
46  std::vector<NameType>&& params,
47  WasmType resultType,
48  std::vector<NameType>&& vars,
49  Expression* body = nullptr) {
50  auto* func = new Function;
51  func->name = name;
52  func->result = resultType;
53  func->body = body;
54 
55  for (auto& param : params) {
56  func->params.push_back(param.type);
57  func->localIndices[param.name] = func->localNames.size();
58  func->localNames.push_back(param.name);
59  }
60  for (auto& var : vars) {
61  func->vars.push_back(var.type);
62  func->localIndices[var.name] = func->localNames.size();
63  func->localNames.push_back(var.name);
64  }
65 
66  return func;
67  }

257  // only ok to add a param if no vars, otherwise indices are invalidated

266  static Index addVar(Function* func, Name name, WasmType type) {
267  // always ok to add a var, it does not affect other indices
268  Index index = func->getNumLocals();
269  if (name.is()) {
270  // if there is a name, apply it, but here we assume all the rest have names too FIXME
271  assert(func->localIndices.size() == func->params.size() + func->vars.size());
272  func->localIndices[name] = index;
273  func->localNames.push_back(name);
274  }

316  // ensures the first node is a block, if it isn't already, and merges in the second,
317  // either as a single element or, if a block, by appending to the first block
318  Block* blockifyMerge(Expression* any, Expression* append) {
319  Block* block = nullptr;
320  if (any) block = any->dynCast<Block>();
321  if (!block) {
322  block = makeBlock(any);
323  } else {
324  assert(!isConcreteWasmType(block->type));
325  }
326  auto* other = append->dynCast<Block>();
327  if (!other) {
328  block->list.push_back(append);
329  } else {

331  block->list.push_back(item);
332  }
333  }
334  block->finalize();
335  return block;

347  // Grab a slice out of a block, replacing it with nops, and returning
348  // either another block with the contents (if more than 1) or a single expression
349  Expression* stealSlice(Block* input, Index from, Index to) {
350  Expression* ret;
351  if (to == from + 1) {
352  // just one
353  ret = input->list[from];
354  } else {
355  auto* block = allocator.alloc<Block>();
356  for (Index i = from; i < to; i++) {
357  block->list.push_back(input->list[i]);
358  }
359  block->finalize();
360  ret = block;
361  }
362  if (to == input->list.size()) {
363  input->list.resize(from);
364  } else {
365  for (Index i = from; i < to; i++) {
366  input->list[i] = allocator.alloc<Nop>();
367  }
368  }
369  input->finalize();
370  return ret;
371  }

384  // returns a replacement with the precise same type, and with
385  // minimal contents. as a replacement, this may reuse the
386  // input node
387  template<typename T>
388  Expression* replaceWithIdenticalType(T* curr) {
389  Literal value;

391  switch (curr->type) {
392  case i32: value = Literal(int32_t(0)); break;
393  case i64: value = Literal(int64_t(0)); break;
394  case f32: value = Literal(float(0)); break;
395  case f64: value = Literal(double(0)); break;
396  case none: return ExpressionManipulator::nop(curr);
