17 // (very) basic linking functionality for s2wasm.
18 // Performs some of the tasks that will eventually be done by a real linker.
19 // Currently can allocate static variables and the stack, lay out memory
20 // and initial segment contents, and process relocations. (In particular, there
21 // is no merging of multiple modules). Currently this is only inteded to turn
22 // a .s file produced by LLVM into a usable wast file.

66  // For now, do not support weak symbols or anything special. Just directly
67  // merge the functions together, and remove any newly-defined functions
68  // from undefinedFunction
69  void merge(SymbolInfo& other) {
70  for (const auto& func : other.implementedFunctions) {
71  undefinedFunctions.erase(func);
72  }
73  implementedFunctions.insert(other.implementedFunctions.begin(),
74  other.implementedFunctions.end());
75  importedObjects.insert(other.importedObjects.begin(),
76  other.importedObjects.end());
77  aliasedSymbols.insert(other.aliasedSymbols.begin(),
78  other.aliasedSymbols.end());
79  }
80  };

197 // Class which performs some linker-like functionality; namely taking an object
198 // file with relocations, laying out the linear memory and segments, and
199 // applying the relocations, resulting in an executable wasm module.
200 class Linker {
201  public:
202  Linker(Address globalBase, Address stackAllocation, Address userInitialMemory,
203  Address userMaxMemory, bool importMemory, bool ignoreUnknownSymbols,
204  Name startFunction, bool debug)
205  : ignoreUnknownSymbols(ignoreUnknownSymbols),
206  startFunction(startFunction),
207  globalBase(globalBase),
208  nextStatic(globalBase),
209  userInitialMemory(userInitialMemory),
210  userMaxMemory(userMaxMemory),
211  importMemory(importMemory),
212  stackAllocation(stackAllocation),
213  debug(debug) {
214  if (userMaxMemory && userMaxMemory < userInitialMemory) {
215  Fatal() << "Specified max memory " << userMaxMemory <<
216  " is < specified initial memory " << userInitialMemory;
217  }
218  if (roundUpToPageSize(userMaxMemory) != userMaxMemory) {
219  Fatal() << "Specified max memory " << userMaxMemory <<
220  " is not a multiple of 64k";
221  }
222  if (roundUpToPageSize(userInitialMemory) != userInitialMemory) {
223  Fatal() << "Specified initial memory " << userInitialMemory <<
224  " is not a multiple of 64k";
225  }
226 
227  // Don't allow anything to be allocated at address 0
228  if (globalBase == 0) nextStatic = 1;
229 
230  // Place the stack pointer at the bottom of the linear memory, to keep its
231  // address small (and thus with a small encoding).
232  placeStackPointer(stackAllocation);
233  // Allocate __dso_handle. For asm.js, emscripten provides this in JS, but
234  // wasm modules can't import data objects. Its value is 0 for the main
235  // executable, which is all we have with static linking. In the future this
236  // can go in a crtbegin or similar file.
237  out.addStatic(4, 4, "__dso_handle");
238  }
