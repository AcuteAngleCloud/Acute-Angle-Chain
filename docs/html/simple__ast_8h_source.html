305  char* parse(char* curr) {
306  #define is_json_space(x) (x == 32 || x == 9 || x == 10 || x == 13) /* space, tab, linefeed/newline, or return */
307  #define skip() { while (*curr && is_json_space(*curr)) curr++; }
308  skip();
309  if (*curr == '"') {
310  // String
311  curr++;
312  char *close = strchr(curr, '"');
313  assert(close);
314  *close = 0; // end this string, and reuse it straight from the input
315  setString(curr);
316  curr = close+1;
317  } else if (*curr == '[') {
318  // Array
319  curr++;
320  skip();
321  setArray();
322  while (*curr != ']') {
323  Ref temp = arena.alloc<Value>();
324  arr->push_back(temp);
325  curr = temp->parse(curr);
326  skip();
327  if (*curr == ']') break;
328  assert(*curr == ',');
329  curr++;
330  skip();
331  }
332  curr++;
333  } else if (*curr == 'n') {
334  // Null
335  assert(strncmp(curr, "null", 4) == 0);
336  setNull();
337  curr += 4;
338  } else if (*curr == 't') {
339  // Bool true
340  assert(strncmp(curr, "true", 4) == 0);
341  setBool(true);
342  curr += 4;
343  } else if (*curr == 'f') {
344  // Bool false
345  assert(strncmp(curr, "false", 5) == 0);
346  setBool(false);
347  curr += 5;
348  } else if (*curr == '{') {
349  // Object
350  curr++;
351  skip();
352  setObject();
353  while (*curr != '}') {
354  assert(*curr == '"');
355  curr++;
356  char *close = strchr(curr, '"');
357  assert(close);
358  *close = 0; // end this string, and reuse it straight from the input
359  IString key(curr);
360  curr = close+1;
361  skip();
362  assert(*curr == ':');
363  curr++;
364  skip();
365  Ref value = arena.alloc<Value>();
366  curr = value->parse(curr);
367  (*obj)[key] = value;
368  skip();
369  if (*curr == '}') break;
370  assert(*curr == ',');
371  curr++;
372  skip();
373  }
374  curr++;
375  } else {
376  // Number
377  char *after;
378  setNumber(strtod(curr, &after));
379  curr = after;
380  }
381  return curr;
382  }

539 struct JSPrinter {
540  static char* numToString(double d, bool finalize=true) {
541  bool neg = d < 0;
542  if (neg) d = -d;
543  // try to emit the fewest necessary characters
544  bool integer = fmod(d, 1) == 0;
545  #define BUFFERSIZE 1000
546  static char full_storage_f[BUFFERSIZE], full_storage_e[BUFFERSIZE]; // f is normal, e is scientific for float, x for integer
547  static char *storage_f = full_storage_f + 1, *storage_e = full_storage_e + 1; // full has one more char, for a possible '-'
548  auto err_f = std::numeric_limits<double>::quiet_NaN();
549  auto err_e = std::numeric_limits<double>::quiet_NaN();
550  for (int e = 0; e <= 1; e++) {
551  char *buffer = e ? storage_e : storage_f;
552  double temp;
553  if (!integer) {
554  static char format[6];
555  for (int i = 0; i <= 18; i++) {
556  format[0] = '%';
557  format[1] = '.';
558  if (i < 10) {
559  format[2] = '0' + i;
560  format[3] = e ? 'e' : 'f';
561  format[4] = 0;
562  } else {
563  format[2] = '1';
564  format[3] = '0' + (i - 10);
565  format[4] = e ? 'e' : 'f';
566  format[5] = 0;
567  }
568  snprintf(buffer, BUFFERSIZE-1, format, d);
569  sscanf(buffer, "%lf", &temp);
570  //errv("%.18f, %.18e => %s => %.18f, %.18e (%d), ", d, d, buffer, temp, temp, temp == d);
571  if (temp == d) break;
572  }
573  } else {
574  // integer
575  assert(d >= 0);
576  if (wasm::isUInteger64(d)) {
577  unsigned long long uu = wasm::toUInteger64(d);
578  bool asHex = e && !finalize;
579  snprintf(buffer, BUFFERSIZE-1, asHex ? "0x%llx" : "%llu", uu);
580  if (asHex) {
581  unsigned long long tempULL;
582  sscanf(buffer, "%llx", &tempULL);
583  temp = (double)tempULL;
584  } else {
585  sscanf(buffer, "%lf", &temp);
586  }
587  } else {
588  // too large for a machine integer, just use floats
589  snprintf(buffer, BUFFERSIZE-1, e ? "%e" : "%.0f", d); // even on integers, e with a dot is useful, e.g. 1.2e+200
590  sscanf(buffer, "%lf", &temp);
591  }
592  //errv("%.18f, %.18e => %s => %.18f, %.18e, %llu (%d)\n", d, d, buffer, temp, temp, uu, temp == d);
593  }
594  (e ? err_e : err_f) = fabs(temp - d);
595  //errv("current attempt: %.18f => %s", d, buffer);
596  //assert(temp == d);
597  char *dot = strchr(buffer, '.');
598  if (dot) {
599  // remove trailing zeros
600  char *end = dot+1;
601  while (*end >= '0' && *end <= '9') end++;
602  end--;
603  while (*end == '0') {
604  char *copy = end;
605  do {
606  copy[0] = copy[1];
607  } while (*copy++ != 0);
608  end--;
609  }
610  //errv("%.18f => %s", d, buffer);
611  // remove preceding zeros
612  while (*buffer == '0') {
613  char *copy = buffer;
614  do {
615  copy[0] = copy[1];
616  } while (*copy++ != 0);
617  }
618  //errv("%.18f ===> %s", d, buffer);
619  } else if (!integer || !e) {
620  // no dot. try to change 12345000 => 12345e3
621  char *end = strchr(buffer, 0);
622  end--;
623  char *test = end;
624  // remove zeros, and also doubles can use at most 24 digits, we can truncate any extras even if not zero
625  while ((*test == '0' || test - buffer > 24) && test > buffer) test--;
626  int num = end - test;
627  if (num >= 3) {
628  test++;
629  test[0] = 'e';
630  if (num < 10) {
631  test[1] = '0' + num;
632  test[2] = 0;
633  } else if (num < 100) {
634  test[1] = '0' + (num / 10);
635  test[2] = '0' + (num % 10);
636  test[3] = 0;
637  } else {
638  assert(num < 1000);
639  test[1] = '0' + (num / 100);
640  test[2] = '0' + (num % 100) / 10;
641  test[3] = '0' + (num % 10);
642  test[4] = 0;
643  }
644  }
645  }
646  //errv("..current attempt: %.18f => %s", d, buffer);
647  }
648  //fprintf(stderr, "options:\n%s\n%s\n (first? %d)\n", storage_e, storage_f, strlen(storage_e) < strlen(storage_f));
649  char *ret;
650  if (err_e == err_f) {
651  ret = strlen(storage_e) < strlen(storage_f) ? storage_e : storage_f;
652  } else {
653  ret = err_e < err_f ? storage_e : storage_f;
654  }
655  if (neg) {
656  ret--; // safe to go back one, there is one more char in full_*
657  *ret = '-';
658  }
659  return ret;
660  }
