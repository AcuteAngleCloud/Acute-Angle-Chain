105  bool runningDefaultOptimizationPasses() {

118  PassRunner getPassRunner(Module& wasm) {
119  PassRunner passRunner(&wasm, passOptions);
120  if (debug) passRunner.setDebug(true);
121  for (auto& pass : passes) {

123  passRunner.addDefaultOptimizationPasses();
