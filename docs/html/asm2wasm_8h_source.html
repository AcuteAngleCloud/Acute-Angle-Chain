178  char* process(char* input) {
179  // emcc --separate-asm modules can look like
180  //
181  // Module["asm"] = (function(global, env, buffer) {
182  // ..
183  // });
184  //
185  // we need to clean that up.
186  if (*input == 'M') {
187  size_t num = strlen(input);
188  while (*input != 'f') {
189  input++;
190  num--;
191  }
192  char *end = input + num - 1;
193  while (*end != '}') {
194  *end = 0;
195  end--;
196  }
197  }
198 
199  // asm.js memory growth uses a quite elaborate pattern. Instead of parsing and
200  // matching it, we do a simpler detection on emscripten's asm.js output format
201  const char* START_FUNCS = "// EMSCRIPTEN_START_FUNCS";
202  char *marker = strstr(input, START_FUNCS);
203  if (marker) {
204  *marker = 0; // look for memory growth code just up to here, as an optimization
205  }
206  char *growthSign = strstr(input, "return true;"); // this can only show up in growth code, as normal asm.js lacks "true"
207  if (growthSign) {
208  memoryGrowth = true;
209  // clean out this function, we don't need it. first where it starts
210  char *growthFuncStart = growthSign;
211  while (*growthFuncStart != '{') growthFuncStart--; // skip body
212  while (*growthFuncStart != '(') growthFuncStart--; // skip params
213  while (*growthFuncStart != ' ') growthFuncStart--; // skip function name
214  while (*growthFuncStart != 'f') growthFuncStart--; // skip 'function'
215  assert(strstr(growthFuncStart, "function ") == growthFuncStart);
216  char *growthFuncEnd = strchr(growthSign, '}');
217  assert(growthFuncEnd > growthFuncStart + 5);
218  growthFuncStart[0] = '/';
219  growthFuncStart[1] = '*';
220  growthFuncEnd--;
221  growthFuncEnd[0] = '*';
222  growthFuncEnd[1] = '/';
223  }
224  if (marker) {
225  *marker = START_FUNCS[0];
226  }
227 
228  // handle debug info, if this build wants that.
229  if (debugInfo) {
230  // asm.js debug info comments look like
231  // ..command..; //@line 4 "tests/hello_world.c"
232  // we convert those into emscripten_debuginfo(file, line)
233  // calls, where the params are indices into a mapping. then
234  // the compiler and optimizer can operate on them. after
235  // that, we can apply the debug info to the wasm node right
236  // before it - this is guaranteed to be correct without opts,
237  // and is usually decently accurate with them.
238  const auto SCALE_FACTOR = 1.25; // an upper bound on how much more space we need as a multiple of the original
239  const auto ADD_FACTOR = 100; // an upper bound on how much we write for each debug info element itself
240  auto size = strlen(input);
241  auto upperBound = Index(size * SCALE_FACTOR) + ADD_FACTOR;
242  char* copy = allocatedCopy = (char*)malloc(upperBound);
243  char* end = copy + upperBound;
244  char* out = copy;
245  std::string DEBUGINFO_INTRINSIC = EMSCRIPTEN_DEBUGINFO.str;
246  auto DEBUGINFO_INTRINSIC_SIZE = DEBUGINFO_INTRINSIC.size();
247  bool seenUseAsm = false;
248  while (input[0]) {
249  if (out + ADD_FACTOR >= end) {
250  Fatal() << "error in handling debug info";
251  }
252  if (startsWith(input, "//@line")) {
253  char* linePos = input + 8;
254  char* lineEnd = strchr(input + 8, ' ');
255  char* filePos = strchr(lineEnd, '"') + 1;
256  char* fileEnd = strchr(filePos, '"');
257  input = fileEnd + 1;
258  *lineEnd = 0;
259  *fileEnd = 0;
260  std::string line = linePos, file = filePos;
261  auto iter = debugInfoFileIndices.find(file);
262  if (iter == debugInfoFileIndices.end()) {
263  Index index = debugInfoFileNames.size();
264  debugInfoFileNames.push_back(file);
265  debugInfoFileIndices[file] = index;
266  }
267  std::string fileIndex = std::to_string(debugInfoFileIndices[file]);
268  // write out the intrinsic
269  strcpy(out, DEBUGINFO_INTRINSIC.c_str());
270  out += DEBUGINFO_INTRINSIC_SIZE;
271  *out++ = '(';
272  strcpy(out, fileIndex.c_str());
273  out += fileIndex.size();
274  *out++ = ',';
275  strcpy(out, line.c_str());
276  out += line.size();
277  *out++ = ')';
278  *out++ = ';';
279  } else if (!seenUseAsm && (startsWith(input, "asm'") || startsWith(input, "asm\""))) {
280  // end of "use asm" or "almost asm"
281  const auto SKIP = 5; // skip the end of "use asm"; (5 chars, a,s,m," or ',;)
282  seenUseAsm = true;
283  memcpy(out, input, SKIP);
284  out += SKIP;
285  input += SKIP;
286  // add a fake import for the intrinsic, so the module validates
287  std::string import = "\n var emscripten_debuginfo = env.emscripten_debuginfo;";
288  strcpy(out, import.c_str());
289  out += import.size();
290  } else {
291  *out++ = *input++;
292  }
293  }
294  if (out >= end) {
295  Fatal() << "error in handling debug info";
296  }
297  *out = 0;
298  input = copy;
299  }
300 
301  return input;
302  }

326  void visitBlock(Block* curr) {
327  // look for a debug info call that is unreachable
328  if (curr->list.size() == 0) return;
329  auto* back = curr->list.back();
330  for (Index i = 1; i < curr->list.size(); i++) {
331  if (checkDebugInfo(curr->list[i]) && !checkDebugInfo(curr->list[i - 1])) {
332  // swap them
333  std::swap(curr->list[i - 1], curr->list[i]);
334  }
335  }
336  if (curr->list.back() != back) {
337  // we changed the last element, update the type
338  curr->finalize();
339  }
340  }

440  void noteImportedFunctionCall(Ref ast, WasmType resultType, CallImport* call) {
441  assert(ast[0] == CALL && ast[1]->isString());
442  IString importName = ast[1]->getIString();
443  auto type = make_unique<FunctionType>();
444  type->name = IString((std::string("type$") + importName.str).c_str(), false); // TODO: make a list of such types
445  type->result = resultType;
446  for (auto* operand : call->operands) {
447  type->params.push_back(operand->type);
448  }
449  // if we already saw this signature, verify it's the same (or else handle that)
450  if (importedFunctionTypes.find(importName) != importedFunctionTypes.end()) {
451  FunctionType* previous = importedFunctionTypes[importName].get();
452  if (*type != *previous) {
453  // merge it in. we'll add on extra 0 parameters for ones not actually used, and upgrade types to
454  // double where there is a conflict (which is ok since in JS, double can contain everything
455  // i32 and f32 can).
456  for (size_t i = 0; i < type->params.size(); i++) {
457  if (previous->params.size() > i) {
458  if (previous->params[i] == none) {
459  previous->params[i] = type->params[i]; // use a more concrete type
460  } else if (previous->params[i] != type->params[i]) {
461  previous->params[i] = f64; // overloaded type, make it a double
462  }
463  } else {
464  previous->params.push_back(type->params[i]); // add a new param
465  }
466  }
467  // we accept none and a concrete type, but two concrete types mean we need to use an f64 to contain anything
468  if (previous->result == none) {
469  previous->result = type->result; // use a more concrete type
470  } else if (previous->result != type->result && type->result != none) {
471  previous->result = f64; // overloaded return type, make it a double
472  }
473  }
474  } else {
475  importedFunctionTypes[importName].swap(type);
476  }
477  }

541  BinaryOp parseAsmBinaryOp(IString op, Ref left, Ref right, Expression* leftWasm, Expression* rightWasm) {
542  WasmType leftType = leftWasm->type;
543  bool isInteger = leftType == WasmType::i32;
544 
545  if (op == PLUS) return isInteger ? BinaryOp::AddInt32 : (leftType == f32 ? BinaryOp::AddFloat32 : BinaryOp::AddFloat64);
546  if (op == MINUS) return isInteger ? BinaryOp::SubInt32 : (leftType == f32 ? BinaryOp::SubFloat32 : BinaryOp::SubFloat64);
547  if (op == MUL) return isInteger ? BinaryOp::MulInt32 : (leftType == f32 ? BinaryOp::MulFloat32 : BinaryOp::MulFloat64);
548  if (op == AND) return BinaryOp::AndInt32;
549  if (op == OR) return BinaryOp::OrInt32;
550  if (op == XOR) return BinaryOp::XorInt32;
551  if (op == LSHIFT) return BinaryOp::ShlInt32;
552  if (op == RSHIFT) return BinaryOp::ShrSInt32;
553  if (op == TRSHIFT) return BinaryOp::ShrUInt32;
554  if (op == EQ) return isInteger ? BinaryOp::EqInt32 : (leftType == f32 ? BinaryOp::EqFloat32 : BinaryOp::EqFloat64);
555  if (op == NE) return isInteger ? BinaryOp::NeInt32 : (leftType == f32 ? BinaryOp::NeFloat32 : BinaryOp::NeFloat64);
556 
557  bool isUnsigned = isUnsignedCoercion(left) || isUnsignedCoercion(right);
558 
559  if (op == DIV) {
560  if (isInteger) {
561  return isUnsigned ? BinaryOp::DivUInt32 : BinaryOp::DivSInt32;
562  }
563  return leftType == f32 ? BinaryOp::DivFloat32 : BinaryOp::DivFloat64;
564  }
565  if (op == MOD) {
566  if (isInteger) {
567  return isUnsigned ? BinaryOp::RemUInt32 : BinaryOp::RemSInt32;
568  }
569  return BinaryOp::RemSInt32; // XXX no floating-point remainder op, this must be handled by the caller
570  }
571  if (op == GE) {
572  if (isInteger) {
573  return isUnsigned ? BinaryOp::GeUInt32 : BinaryOp::GeSInt32;
574  }
575  return leftType == f32 ? BinaryOp::GeFloat32 : BinaryOp::GeFloat64;
576  }
577  if (op == GT) {
578  if (isInteger) {
579  return isUnsigned ? BinaryOp::GtUInt32 : BinaryOp::GtSInt32;
580  }
581  return leftType == f32 ? BinaryOp::GtFloat32 : BinaryOp::GtFloat64;
582  }
583  if (op == LE) {
584  if (isInteger) {
585  return isUnsigned ? BinaryOp::LeUInt32 : BinaryOp::LeSInt32;
586  }
587  return leftType == f32 ? BinaryOp::LeFloat32 : BinaryOp::LeFloat64;
588  }
589  if (op == LT) {
590  if (isInteger) {
591  return isUnsigned ? BinaryOp::LtUInt32 : BinaryOp::LtSInt32;
592  }
593  return leftType == f32 ? BinaryOp::LtFloat32 : BinaryOp::LtFloat64;
594  }
595  abort_on("bad wasm binary op", op);
596  abort(); // avoid warning
597  }

613  Literal checkLiteral(Ref ast, bool rawIsInteger = true) {
614  if (ast->isNumber()) {
615  if (rawIsInteger) {
616  return Literal((int32_t)ast->getInteger());
617  } else {
618  return Literal(ast->getNumber());
619  }
620  } else if (ast->isArray(UNARY_PREFIX)) {

622  return Literal((double)ast[2]->getNumber());
623  }

791  Expression* makeTrappingFloatToInt(bool signed_, Expression* value) {
792  if (trapMode == TrapMode::Allow) {
793  auto ret = allocator.alloc<Unary>();
794  ret->value = value;
795  bool isF64 = ret->value->type == f64;
796  if (signed_) {
797  ret->op = isF64 ? TruncSFloat64ToInt32 : TruncSFloat32ToInt32;
798  } else {
799  ret->op = isF64 ? TruncUFloat64ToInt32 : TruncUFloat32ToInt32;
800  }
801  ret->type = WasmType::i32;
802  return ret;
803  }
804  // WebAssembly traps on float-to-int overflows, but asm.js wouldn't, so we must do something
805  // First, normalize input to f64
806  auto input = value;
807  if (input->type == f32) {
808  auto conv = allocator.alloc<Unary>();
809  conv->op = PromoteFloat32;
810  conv->value = input;
811  conv->type = WasmType::f64;
812  input = conv;
813  }
814  // We can handle this in one of two ways: clamping, which is fast, or JS, which
815  // is precisely like JS but in order to do that we do a slow ffi
816  if (trapMode == TrapMode::JS) {
817  // WebAssembly traps on float-to-int overflows, but asm.js wouldn't, so we must emulate that
818  CallImport *ret = allocator.alloc<CallImport>();
819  ret->target = F64_TO_INT;
820  ret->operands.push_back(input);
821  ret->type = i32;
822  static bool addedImport = false;
823  if (!addedImport) {
824  addedImport = true;

826  import->name = F64_TO_INT;
827  import->module = ASM2WASM;
828  import->base = F64_TO_INT;
829  import->functionType = ensureFunctionType("id", &wasm)->name;
830  import->kind = ExternalKind::Function;
831  wasm.addImport(import);
832  }
833  return ret;
834  }
835  assert(trapMode == TrapMode::Clamp);
836  Call *ret = allocator.alloc<Call>();
837  ret->target = F64_TO_INT;
838  ret->operands.push_back(input);
839  ret->type = i32;
840  static bool added = false;
841  if (!added) {
842  added = true;
843  auto func = new Function;
844  func->name = ret->target;
845  func->params.push_back(f64);
846  func->result = i32;
847  func->body = builder.makeUnary(TruncSFloat64ToInt32,
848  builder.makeGetLocal(0, f64)
849  );
850  // too small XXX this is different than asm.js, which does frem. here we clamp, which is much simpler/faster, and similar to native builds
851  func->body = builder.makeIf(
852  builder.makeBinary(LeFloat64,
853  builder.makeGetLocal(0, f64),
854  builder.makeConst(Literal(double(std::numeric_limits<int32_t>::min()) - 1))
855  ),
856  builder.makeConst(Literal(int32_t(std::numeric_limits<int32_t>::min()))),
857  func->body
858  );
859  // too big XXX see above
860  func->body = builder.makeIf(
861  builder.makeBinary(GeFloat64,
862  builder.makeGetLocal(0, f64),
863  builder.makeConst(Literal(double(std::numeric_limits<int32_t>::max()) + 1))
864  ),
865  builder.makeConst(Literal(int32_t(std::numeric_limits<int32_t>::min()))), // NB: min here as well. anything out of range => to the min
866  func->body
867  );
868  // nan
869  func->body = builder.makeIf(
870  builder.makeBinary(NeFloat64,
871  builder.makeGetLocal(0, f64),
872  builder.makeGetLocal(0, f64)
873  ),
874  builder.makeConst(Literal(int32_t(std::numeric_limits<int32_t>::min()))), // NB: min here as well. anything invalid => to the min
875  func->body
876  );
877  wasm.addFunction(func);
878  }
