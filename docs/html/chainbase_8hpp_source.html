316

325  {

327  }
328  _next_id = head.old_next_id;
329

345  void squash()
346  {
347  if( !enabled() ) return;
348  if( _stack.size() == 1 ) {
349  _stack.pop_front();
350  return;
351  }
352 
353  auto& state = _stack.back();
354  auto& prev_state = _stack[_stack.size()-2];
355 
356  // An object's relationship to a state can be:
357  // in new_ids : new
358  // in old_values (was=X) : upd(was=X)
359  // in removed (was=X) : del(was=X)
360  // not in any of above : nop
361  //
362  // When merging A=prev_state and B=state we have a 4x4 matrix of all possibilities:
363  //
364  // |--------------------- B ----------------------|
365  //
366  // +------------+------------+------------+------------+
367  // | new | upd(was=Y) | del(was=Y) | nop |
368  // +------------+------------+------------+------------+------------+
369  // / | new | N/A | new A| nop C| new A|
370  // | +------------+------------+------------+------------+------------+
371  // | | upd(was=X) | N/A | upd(was=X)A| del(was=X)C| upd(was=X)A|
372  // A +------------+------------+------------+------------+------------+
373  // | | del(was=X) | N/A | N/A | N/A | del(was=X)A|
374  // | +------------+------------+------------+------------+------------+
375  // \ | nop | new B| upd(was=Y)B| del(was=Y)B| nop AB|
376  // +------------+------------+------------+------------+------------+
377  //
378  // Each entry was composed by labelling what should occur in the given case.
379  //
380  // Type A means the composition of states contains the same entry as the first of the two merged states for that object.
381  // Type B means the composition of states contains the same entry as the second of the two merged states for that object.
382  // Type C means the composition of states contains an entry different from either of the merged states for that object.
383  // Type N/A means the composition of states violates causal timing.
384  // Type AB means both type A and type B simultaneously.
385  //
386  // The merge() operation is defined as modifying prev_state in-place to be the state object which represents the composition of
387  // state A and B.
388  //
389  // Type A (and AB) can be implemented as a no-op; prev_state already contains the correct value for the merged state.
390  // Type B (and AB) can be implemented by copying from state to prev_state.
391  // Type C needs special case-by-case logic.
392  // Type N/A can be ignored or assert(false) as it can only occur if prev_state and state have illegal values
393  // (a serious logic error which should never happen).
394  //
395 
396  // We can only be outside type A/AB (the nop path) if B is not nop, so it suffices to iterate through B's three containers.
397 
398  for( const auto& item : state.old_values )
399  {
400  if( prev_state.new_ids.find( item.second.id ) != prev_state.new_ids.end() )
401  {
402  // new+upd -> new, type A
403  continue;
404  }
405  if( prev_state.old_values.find( item.second.id ) != prev_state.old_values.end() )
406  {
407  // upd(was=X) + upd(was=Y) -> upd(was=X), type A
408  continue;
409  }
410  // del+upd -> N/A
411  assert( prev_state.removed_values.find(item.second.id) == prev_state.removed_values.end() );
412  // nop+upd(was=Y) -> upd(was=Y), type B
413  prev_state.old_values.emplace( std::move(item) );
414  }
415 
416  // *+new, but we assume the N/A cases don't happen, leaving type B nop+new -> new
417  for( auto id : state.new_ids )
418  prev_state.new_ids.insert(id);
419 
420  // *+del
421  for( auto& obj : state.removed_values )
422  {
423  if( prev_state.new_ids.find(obj.second.id) != prev_state.new_ids.end() )
424  {
425  // new + del -> nop (type C)
426  prev_state.new_ids.erase(obj.second.id);
427  continue;
428  }
429  auto it = prev_state.old_values.find(obj.second.id);
430  if( it != prev_state.old_values.end() )
431  {
432  // upd(was=X) + del(was=Y) -> del(was=X)
433  prev_state.removed_values.emplace( std::move(*it) );
434  prev_state.old_values.erase(obj.second.id);
435  continue;
436  }
437  // del + del -> N/A
438  assert( prev_state.removed_values.find( obj.second.id ) == prev_state.removed_values.end() );
439  // nop + del(was=Y) -> del(was=Y)
440  prev_state.removed_values.emplace( std::move(obj) ); //[obj.second->id] = std::move(obj.second);
441  }
442 
443  _stack.pop_back();
444  --_revision;
445  }
