157  static bool isSpace(char x) { return x == 32 || x == 9 || x == 10 || x == 13; } /* space, tab, linefeed/newline, or return */
158  static void skipSpace(char*& curr) {
159  while (*curr) {
160  if (isSpace(*curr)) {
161  curr++;
162  continue;
163  }
164  if (curr[0] == '/' && curr[1] == '/') {
165  curr += 2;
166  while (*curr && *curr != '\n') curr++;
167  if (*curr) curr++;
168  continue;
169  }
170  if (curr[0] == '/' && curr[1] == '*') {
171  curr += 2;
172  while (*curr && (curr[0] != '*' || curr[1] != '/')) curr++;
173  curr += 2;
174  continue;
175  }
176  return;
177  }
178  }

195  struct Frag {
196 #ifndef _MSC_VER // MSVC does not allow unrestricted unions: http://www.open-std.org/jtc1/sc22/wg21/docs/papers/2008/n2544.pdf
197  union {
198 #endif
199  IString str;
200  double num;
201 #ifndef _MSC_VER
202  };
203 #endif
204  int size;
205  FragType type;
206 
207  bool isNumber() const {
208  return type == INT || type == DOUBLE;
209  }
210 
211  explicit Frag(char* src) {
212  char *start = src;
213  if (isIdentInit(*src)) {
214  // read an identifier or a keyword
215  src++;
216  while (isIdentPart(*src)) {
217  src++;
218  }
219  if (*src == 0) {
220  str.set(start);
221  } else {
222  char temp = *src;
223  *src = 0;
224  str.set(start, false);
225  *src = temp;
226  }
227  type = keywords.has(str) ? KEYWORD : IDENT;
228  } else if (isDigit(*src) || (src[0] == '.' && isDigit(src[1]))) {
229  if (src[0] == '0' && (src[1] == 'x' || src[1] == 'X')) {
230  // Explicitly parse hex numbers of form "0x...", because strtod
231  // supports hex number strings only in C++11, and Visual Studio 2013 does
232  // not yet support that functionality.
233  src += 2;
234  num = 0;
235  while (1) {
236  if (*src >= '0' && *src <= '9') { num *= 16; num += *src - '0'; }
237  else if (*src >= 'a' && *src <= 'f') { num *= 16; num += *src - 'a' + 10; }
238  else if (*src >= 'A' && *src <= 'F') { num *= 16; num += *src - 'A' + 10; }
239  else break;
240  src++;
241  }
242  } else {
243  num = strtod(start, &src);
244  }
245  // asm.js must have a '.' for double values. however, we also tolerate
246  // uglify's tendency to emit without a '.' (and fix it later with a +).
247  // for valid asm.js input, the '.' should be enough, and for uglify
248  // in the emscripten optimizer pipeline, we use simple_ast where INT/DOUBLE
249  // is quite the same at this point anyhow
250  type = (std::find(start, src, '.') == src &&
251  (wasm::isSInteger32(num) || wasm::isUInteger32(num)))
252  ? INT
253  : DOUBLE;
254  assert(src > start);
255  } else if (hasChar(OPERATOR_INITS, *src)) {
256  switch (*src) {
257  case '!': str = src[1] == '=' ? NE : L_NOT; break;
258  case '%': str = MOD; break;
259  case '&': str = AND; break;
260  case '*': str = MUL; break;
261  case '+': str = PLUS; break;
262  case ',': str = COMMA; break;
263  case '-': str = MINUS; break;
264  case '.': str = PERIOD; break;
265  case '/': str = DIV; break;
266  case ':': str = COLON; break;
267  case '<': str = src[1] == '<' ? LSHIFT : (src[1] == '=' ? LE : LT); break;
268  case '=': str = src[1] == '=' ? EQ : SET; break;
269  case '>': str = src[1] == '>' ? (src[2] == '>' ? TRSHIFT : RSHIFT) : (src[1] == '=' ? GE : GT); break;
270  case '?': str = QUESTION; break;
271  case '^': str = XOR; break;
272  case '|': str = OR; break;
273  case '~': str = B_NOT; break;
274  default: abort();
275  }
276  size = strlen(str.str);
277 #ifndef NDEBUG
278  char temp = start[size];
279  start[size] = 0;
280  assert(strcmp(str.str, start) == 0);
281  start[size] = temp;
282 #endif
283  type = OPERATOR;
284  return;
285  } else if (hasChar(SEPARATORS, *src)) {
286  type = SEPARATOR;
287  char temp = src[1];
288  src[1] = 0;
289  str.set(src, false);
290  src[1] = temp;
291  src++;
292  } else if (*src == '"' || *src == '\'') {
293  char *end = strchr(src+1, *src);
294  *end = 0;
295  str.set(src+1);
296  src = end+1;
297  type = STRING;
298  } else {
299  dump("frag parsing", src);
300  abort();
301  }
302  size = src - start;
303  }
304  };

334  // Parses an element in a list of such elements, e.g. list of statements in a block, or list of parameters in a call
335  NodeRef parseElement(char*& src, const char* seps=";") {
336  //dump("parseElement", src);
337  skipSpace(src);
338  Frag frag(src);
339  src += frag.size;
340  switch (frag.type) {
341  case KEYWORD: {
342  return parseAfterKeyword(frag, src, seps);
343  }
344  case IDENT: {
345  return parseAfterIdent(frag, src, seps);
346  }
347  case STRING:
348  case INT:
349  case DOUBLE: {
350  return parseExpression(parseFrag(frag), src, seps);
351  }
352  case SEPARATOR: {
353  if (frag.str == OPEN_PAREN) return parseExpression(parseAfterParen(src), src, seps);
354  if (frag.str == OPEN_BRACE) return parseExpression(parseAfterBrace(src), src, seps);
355  if (frag.str == OPEN_CURLY) return parseExpression(parseAfterCurly(src), src, seps);
356  abort();
357  }
358  case OPERATOR: {
359  return parseExpression(frag.str, src, seps);
360  }
361  default: /* dump("parseElement", src); printf("bad frag type: %d\n", frag.type); */ abort();
362  }
363  return nullptr;
364  }

366  NodeRef parseFrag(Frag& frag) {
367  switch (frag.type) {
368  case IDENT: return Builder::makeName(frag.str);
369  case STRING: return Builder::makeString(frag.str);
370  case INT: return Builder::makeInt(uint32_t(frag.num));
371  case DOUBLE: return Builder::makeDouble(frag.num);
372  default: abort();
373  }
374  return nullptr;
375  }

377  NodeRef parseAfterKeyword(Frag& frag, char*& src, const char* seps) {
378  skipSpace(src);
379  if (frag.str == FUNCTION) return parseFunction(src, seps);
380  else if (frag.str == VAR) return parseVar(src, seps, false);
381  else if (frag.str == CONST) return parseVar(src, seps, true);
382  else if (frag.str == RETURN) return parseReturn(src, seps);
383  else if (frag.str == IF) return parseIf(src, seps);
384  else if (frag.str == DO) return parseDo(src, seps);
385  else if (frag.str == WHILE) return parseWhile(src, seps);
386  else if (frag.str == BREAK) return parseBreak(src, seps);
387  else if (frag.str == CONTINUE) return parseContinue(src, seps);
388  else if (frag.str == SWITCH) return parseSwitch(src, seps);
389  else if (frag.str == NEW) return parseNew(src, seps);
390  else if (frag.str == FOR) return parseFor(src, seps);
391  dump(frag.str.str, src);
392  abort();
393  return nullptr;
394  }

396  NodeRef parseFunction(char*& src, const char* seps) {
397  Frag name(src);
398  if (name.type == IDENT) {
399  src += name.size;
400  } else {
401  assert(name.type == SEPARATOR && name.str[0] == '(');
402  name.str = IString();
403  }
404  NodeRef ret = Builder::makeFunction(name.str);
405  skipSpace(src);
406  assert(*src == '(');
407  src++;
408  while (1) {
409  skipSpace(src);
410  if (*src == ')') break;
411  Frag arg(src);
412  assert(arg.type == IDENT);
413  src += arg.size;
414  Builder::appendArgumentToFunction(ret, arg.str);
415  skipSpace(src);
416  if (*src == ')') break;
417  if (*src == ',') {
418  src++;
419  continue;
420  }
421  abort();
422  }
423  src++;
424  Builder::setBlockContent(ret, parseBracketedBlock(src));
425  // TODO: parse expression?
426  return ret;
427  }

447  if (*src == ',') {
448  src++;
449  continue;
450  }
451  abort();
452  }
453  src++;
454  return ret;
455  }

467  NodeRef condition = parseParenned(src);
468  NodeRef ifTrue = parseMaybeBracketed(src, seps);

531  NodeRef parseSwitch(char*& src, const char* seps) {
532  NodeRef ret = Builder::makeSwitch(parseParenned(src));
533  skipSpace(src);
534  assert(*src == '{');
535  src++;
536  while (1) {
537  // find all cases and possibly a default
538  skipSpace(src);
539  if (*src == '}') break;
540  Frag next(src);
541  if (next.type == KEYWORD) {
542  if (next.str == CASE) {
543  src += next.size;
544  skipSpace(src);
545  NodeRef arg;
546  Frag value(src);
547  if (value.isNumber()) {
548  arg = parseFrag(value);
549  src += value.size;
550  } else if (value.type == OPERATOR) {
551  // negative number
552  assert(value.str == MINUS);
553  src += value.size;
554  skipSpace(src);
555  Frag value2(src);
556  assert(value2.isNumber());
557  arg = Builder::makePrefix(MINUS, parseFrag(value2));
558  src += value2.size;
559  } else {
560  // identifier and function call
561  assert(value.type == IDENT);
562  src += value.size;
563  skipSpace(src);
564  arg = parseCall(parseFrag(value), src);
565  }
566  Builder::appendCaseToSwitch(ret, arg);
567  skipSpace(src);
568  assert(*src == ':');
569  src++;
570  continue;
571  } else if (next.str == DEFAULT) {
572  src += next.size;
573  Builder::appendDefaultToSwitch(ret);
574  skipSpace(src);
575  assert(*src == ':');
576  src++;
577  continue;
578  }
579  // otherwise, may be some keyword that happens to start a block (e.g. case 1: _return_ 5)
580  }
581  // not case X: or default: or }, so must be some code
582  skipSpace(src);
583  bool explicitBlock = *src == '{';
584  NodeRef subBlock = explicitBlock ? parseBracketedBlock(src) : parseBlock(src, ";}", CASE, DEFAULT);
585  Builder::appendCodeToSwitch(ret, subBlock, explicitBlock);
586  }
587  skipSpace(src);
588  assert(*src == '}');
589  src++;
590  return ret;
591  }

597  NodeRef parseAfterIdent(Frag& frag, char*& src, const char* seps) {
598  skipSpace(src);
599  if (*src == '(') return parseExpression(parseCall(parseFrag(frag), src), src, seps);
600  if (*src == '[') return parseExpression(parseIndexing(parseFrag(frag), src), src, seps);
601  if (*src == ':' && expressionPartsStack.back().size() == 0) {
602  src++;
603  skipSpace(src);
604  NodeRef inner;
605  if (*src == '{') { // context lets us know this is not an object, but a block
606  inner = parseBracketedBlock(src);
607  } else {
608  inner = parseElement(src, seps);
609  }
610  return Builder::makeLabel(frag.str, inner);
611  }
612  if (*src == '.') return parseExpression(parseDotting(parseFrag(frag), src), src, seps);
613  return parseExpression(parseFrag(frag), src, seps);
614  }

616  NodeRef parseCall(NodeRef target, char*& src) {
617  expressionPartsStack.resize(expressionPartsStack.size()+1);
618  assert(*src == '(');
619  src++;
620  NodeRef ret = Builder::makeCall(target);
621  while (1) {
622  skipSpace(src);
623  if (*src == ')') break;
624  Builder::appendToCall(ret, parseElement(src, ",)"));
625  skipSpace(src);
626  if (*src == ')') break;
627  if (*src == ',') {
628  src++;
629  continue;
630  }
631  abort();
632  }
633  src++;
634  assert(expressionPartsStack.back().size() == 0);
635  expressionPartsStack.pop_back();
636  return ret;
637  }

743  NodeRef parseExpression(ExpressionElement initial, char*&src, const char* seps) {
744  //dump("parseExpression", src);
745  ExpressionParts& parts = expressionPartsStack.back();
746  skipSpace(src);
747  if (*src == 0 || hasChar(seps, *src)) {
748  if (parts.size() > 0) {
749  parts.push_back(initial); // cherry on top of the cake
750  }
751  return initial.getNode();
752  }
753  bool top = parts.size() == 0;
754  if (initial.isNode) {
755  Frag next(src);
756  if (next.type == OPERATOR) {
757  parts.push_back(initial);
758  src += next.size;
759  parts.push_back(next.str);
760  } else {
761  if (*src == '(') {
762  initial = parseCall(initial.getNode(), src);
763  } else if (*src == '[') {
764  initial = parseIndexing(initial.getNode(), src);
765  } else {
766  dump("bad parseExpression state", src);
767  abort();
768  }
769  return parseExpression(initial, src, seps);
770  }
771  } else {
772  parts.push_back(initial);
773  }
774  NodeRef last = parseElement(src, seps);
775  if (!top) return last;
776  {
777  ExpressionParts& parts = expressionPartsStack.back(); // |parts| may have been invalidated by that call
778  // we are the toplevel. sort it all out
779  // collapse right to left, highest priority first
780  //dumpParts(parts, 0);
781  for (auto& ops : operatorClasses) {
782  if (ops.rtl) {
783  // right to left
784  for (int i = parts.size()-1; i >= 0; i--) {
785  if (parts[i].isNode) continue;
786  IString op = parts[i].getOp();
787  if (!ops.ops.has(op)) continue;
788  if (ops.type == OperatorClass::Binary && i > 0 && i < (int)parts.size()-1) {
789  parts[i] = makeBinary(parts[i-1].getNode(), op, parts[i+1].getNode());
790  parts.erase(parts.begin() + i + 1);
791  parts.erase(parts.begin() + i - 1);
792  } else if (ops.type == OperatorClass::Prefix && i < (int)parts.size()-1) {
793  if (i > 0 && parts[i-1].isNode) continue; // cannot apply prefix operator if it would join two nodes
794  parts[i] = Builder::makePrefix(op, parts[i+1].getNode());
795  parts.erase(parts.begin() + i + 1);
796  } else if (ops.type == OperatorClass::Tertiary) {
797  // we must be at X ? Y : Z
798  // ^
799  //dumpParts(parts, i);
800  if (op != COLON) continue;
801  assert(i < (int)parts.size()-1 && i >= 3);
802  if (parts[i-2].getOp() != QUESTION) continue; // e.g. x ? y ? 1 : 0 : 2
803  parts[i-3] = Builder::makeConditional(parts[i-3].getNode(), parts[i-1].getNode(), parts[i+1].getNode());
804  parts.erase(parts.begin() + i - 2, parts.begin() + i + 2);
805  i = parts.size(); // basically a reset, due to things like x ? y ? 1 : 0 : 2
806  } // TODO: postfix
807  }
808  } else {
809  // left to right
810  for (int i = 0; i < (int)parts.size(); i++) {
811  if (parts[i].isNode) continue;
812  IString op = parts[i].getOp();
813  if (!ops.ops.has(op)) continue;
814  if (ops.type == OperatorClass::Binary && i > 0 && i < (int)parts.size()-1) {
815  parts[i] = makeBinary(parts[i-1].getNode(), op, parts[i+1].getNode());
816  parts.erase(parts.begin() + i + 1);
817  parts.erase(parts.begin() + i - 1);
818  i--;
819  } else if (ops.type == OperatorClass::Prefix && i < (int)parts.size()-1) {
820  if (i > 0 && parts[i-1].isNode) continue; // cannot apply prefix operator if it would join two nodes
821  parts[i] = Builder::makePrefix(op, parts[i+1].getNode());
822  parts.erase(parts.begin() + i + 1);
823  i = std::max(i-2, 0); // allow a previous prefix operator to cascade
824  } // TODO: tertiary, postfix
825  }
826  }
827  }
828  assert(parts.size() == 1);
829  NodeRef ret = parts[0].getNode();
830  parts.clear();
831  return ret;
832  }

836  NodeRef parseBlock(char*& src, const char* seps=";", IString keywordSep1=IString(), IString keywordSep2=IString()) {
837  NodeRef block = Builder::makeBlock();
838  //dump("parseBlock", src);
839  while (1) {
840  skipSpace(src);
841  if (*src == 0) break;
842  if (*src == ';') {
843  src++; // skip a statement in this block
844  continue;
845  }
846  if (hasChar(seps, *src)) break;
847  if (!!keywordSep1) {
848  Frag next(src);
849  if (next.type == KEYWORD && next.str == keywordSep1) break;
850  }
851  if (!!keywordSep2) {
852  Frag next(src);
853  if (next.type == KEYWORD && next.str == keywordSep2) break;
854  }
855  NodeRef element = parseElementOrStatement(src, seps);
856  Builder::appendToBlock(block, element);
857  }
858  return block;
859  }
