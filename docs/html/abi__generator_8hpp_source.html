236  void MacroExpands (const Token &token, const MacroDefinition &md, SourceRange range, const MacroArgs *args) override {
237 
238  auto* id = token.getIdentifierInfo();
239  if( id == nullptr ) return;
240  if( id->getName() != "AACIO_ABI" ) return;
241 
242  const auto& sm = compiler_instance.getSourceManager();
243  auto file_name = sm.getFilename(range.getBegin());
244  if ( !act.abi_context.empty() && !file_name.startswith(act.abi_context) ) {
245  return;
246  }
247 
248  ABI_ASSERT( md.getMacroInfo()->getNumArgs() == 2 );
249 
250  clang::SourceLocation b(range.getBegin()), _e(range.getEnd());
251  clang::SourceLocation e(clang::Lexer::getLocForEndOfToken(_e, 0, sm, compiler_instance.getLangOpts()));
252  auto macrostr = string(sm.getCharacterData(b), sm.getCharacterData(e)-sm.getCharacterData(b));
253 
254  regex r(R"(AACIO_ABI\s*\(\s*(.+?)\s*,((?:.+?)*)\s*\))");
255  smatch smatch;
256  auto res = regex_search(macrostr, smatch, r);
257  ABI_ASSERT( res );
258 
259  act.contract = smatch[1].str();
260 
261  auto actions_str = smatch[2].str();
262  boost::trim(actions_str);
263  actions_str = actions_str.substr(1);
264  actions_str.pop_back();
265  boost::remove_erase_if(actions_str, boost::is_any_of(" ("));
266 
267  boost::split(act.actions, actions_str, boost::is_any_of(")"));
268  }
269  };
