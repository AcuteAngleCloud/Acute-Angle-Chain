25 //
26 // Machine learning using a genetic algorithm.
27 //
28 // The Genome class is the element on which to learn. It must
29 // implement the following:
30 //
31 // * Fitness getFitness(); - calculate how good this item is.
32 //
33 // The Generator must implement the following:
34 //
35 // * Genome* makeRandom(); - make a random element
36 // * Genome* makeMixture(Genome* one, Genome* two); - make a new element by mixing two
37 //
38 // Fitness is the type of the fitness values, e.g. uint32_t. More is better.
39 //
40 // Typical usage of this class is to run call runGeneration(), check the best
41 // quality using getBest()->getFitness(), and do that repeatedly until the
42 // fitness is good enough. Then acquireBest() to get ownership of the best,
43 // and the learner can be discarded (with all the rest of the population
44 // cleaned up).
45 
46 template<typename Genome, typename Fitness, typename Generator>
47 class GeneticLearner {
48  Generator& generator;
49 
50  typedef std::unique_ptr<Genome> unique_ptr;
51  std::vector<unique_ptr> population;
52 
53  void sort() {
54  std::sort(population.begin(), population.end(), [](const unique_ptr& left, const unique_ptr& right) {
55  return left->getFitness() > right->getFitness();
56  });
57  }
58 
59  std::mt19937 noise;
60 
61  size_t randomIndex() {
62  return noise() % population.size();
63  }
64 
65 public:
66  GeneticLearner(Generator& generator, size_t size) : generator(generator), noise(1337) {
67  population.resize(size);
68  for (size_t i = 0; i < size; i++) {
69  population[i] = unique_ptr(generator.makeRandom());
70  }
71  sort();
72  }
73 
74  Genome* getBest() {
75  return population[0].get();
76  }
77 
78  unique_ptr acquireBest() {
79  return population[0];
80  }
81 
82  void runGeneration() {
83  size_t size = population.size();
84 
85  // we have a mix of promoted from the last generation, mixed from the last generation, and random
86  const size_t promoted = (25 * size) / 100;
87  const size_t mixed = (50 * size) / 100;
88 
89  // promoted just stay in place
90  // mixtures are computed, then added back in (as we still need them as we work)
91  std::vector<unique_ptr> mixtures;
92  mixtures.resize(mixed);
93  for (size_t i = 0; i < mixed; i++) {
94  mixtures[i] = unique_ptr(generator.makeMixture(population[randomIndex()].get(), population[randomIndex()].get()));
95  }
96  for (size_t i = 0; i < mixed; i++) {
97  population[promoted + i].swap(mixtures[i]);
98  }
99  // TODO: de-duplicate at this point
100  // randoms fill in the test
101  for (size_t i = promoted + mixed; i < size; i++) {
102  population[i] = unique_ptr(generator.makeRandom());
103  }
104 
105  sort();
106  }
107 };
