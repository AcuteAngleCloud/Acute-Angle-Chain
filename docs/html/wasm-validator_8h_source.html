17 //
18 // Simple WebAssembly module validator.
19 //
20 // There are some options regarding how to validate:
21 //
22 // * validateWeb: The Web platform doesn't have i64 values, so it is illegal
23 // to import or export such a value. When this option is set,
24 // such imports/exports are validation errors.
25 //
26 // * validateGlobally: Binaryen supports building modules in parallel, which
27 // means you can add and optimize a function before the
28 // module is complete, for example, you can add function A
29 // with a call to function B before function B exists.
30 // When validateGlobally is disabled, we don't look at
31 // global correctness, and instead only check inside
32 // each function (so in the example above we wouldn't care
33 // about function B not existing yet, but we would care
34 // if e.g. inside function A an i32.add receives an i64).
35 //

78  bool validate(Module& module, bool validateWeb_ = false, bool validateGlobally_ = true) {
79  validateWeb = validateWeb_;
80  validateGlobally = validateGlobally_;
81  // wasm logic validation
82  walkModule(&module);
83  // validate additional internal IR details when in pass-debug mode
84  if (PassRunner::getPassDebug()) {
85  validateBinaryenIR(module);
86  }
87  // print if an error occurred
88  if (!valid) {
89  WasmPrinter::printModule(&module, std::cerr);
90  }
91  return valid;
92  }

96  static void visitPreBlock(WasmValidator* self, Expression** currp) {
97  auto* curr = (*currp)->cast<Block>();
98  if (curr->name.is()) self->breakTargets[curr->name].push_back(curr);
99  }

101  void visitBlock(Block *curr) {
102  // if we are break'ed to, then the value must be right for us
103  if (curr->name.is()) {
104  noteLabelName(curr->name);
105  if (breakInfos.count(curr) > 0) {
106  auto& info = breakInfos[curr];
107  if (isConcreteWasmType(curr->type)) {
108  shouldBeTrue(info.arity != 0, curr, "break arities must be > 0 if block has a value");
109  } else {
110  shouldBeTrue(info.arity == 0, curr, "break arities must be 0 if block has no value");
111  }
112  // none or unreachable means a poison value that we should ignore - if consumed, it will error
113  if (isConcreteWasmType(info.type) && isConcreteWasmType(curr->type)) {
114  shouldBeEqual(curr->type, info.type, curr, "block+breaks must have right type if breaks return a value");
115  }
116  if (isConcreteWasmType(curr->type) && info.arity && info.type != unreachable) {
117  shouldBeEqual(curr->type, info.type, curr, "block+breaks must have right type if breaks have arity");
118  }
119  shouldBeTrue(info.arity != Index(-1), curr, "break arities must match");
120  if (curr->list.size() > 0) {
121  auto last = curr->list.back()->type;
122  if (isConcreteWasmType(last) && info.type != unreachable) {
123  shouldBeEqual(last, info.type, curr, "block+breaks must have right type if block ends with a reachable value");
124  }
125  if (last == none) {
126  shouldBeTrue(info.arity == Index(0), curr, "if block ends with a none, breaks cannot send a value of any type");
127  }
128  }
129  }
130  breakTargets[curr->name].pop_back();
131  }
132  if (curr->list.size() > 1) {
133  for (Index i = 0; i < curr->list.size() - 1; i++) {
134  if (!shouldBeTrue(!isConcreteWasmType(curr->list[i]->type), curr, "non-final block elements returning a value must be drop()ed (binaryen's autodrop option might help you)")) {
135  std::cerr << "(on index " << i << ":\n" << curr->list[i] << "\n), type: " << curr->list[i]->type << "\n";
136  }
137  }
138  }
139  if (curr->list.size() > 0) {
140  auto backType = curr->list.back()->type;
141  if (!isConcreteWasmType(curr->type)) {
142  if (isConcreteWasmType(backType)) {
143  shouldBeTrue(curr->type == unreachable, curr, "block with no value and a last element with a value must be unreachable");
144  }
145  } else {
146  if (isConcreteWasmType(backType)) {
147  shouldBeEqual(curr->type, backType, curr, "block with value and last element with value must match types");
148  } else {
149  shouldBeUnequal(backType, none, curr, "block with value must not have last element that is none");
150  }
151  }
152  }
153  if (isConcreteWasmType(curr->type)) {
154  shouldBeTrue(curr->list.size() > 0, curr, "block with a value must not be empty");
155  }
156  }

158  static void visitPreLoop(WasmValidator* self, Expression** currp) {
159  auto* curr = (*currp)->cast<Loop>();
160  if (curr->name.is()) self->breakTargets[curr->name].push_back(curr);
161  }

163  void visitLoop(Loop *curr) {
164  if (curr->name.is()) {
165  noteLabelName(curr->name);
166  breakTargets[curr->name].pop_back();
167  if (breakInfos.count(curr) > 0) {
168  auto& info = breakInfos[curr];
169  shouldBeEqual(info.arity, Index(0), curr, "breaks to a loop cannot pass a value");
170  }
171  }
172  if (curr->type == none) {
173  shouldBeFalse(isConcreteWasmType(curr->body->type), curr, "bad body for a loop that has no value");
174  }
175  }

177  void visitIf(If *curr) {
178  shouldBeTrue(curr->condition->type == unreachable || curr->condition->type == i32, curr, "if condition must be valid");
179  if (!curr->ifFalse) {
180  shouldBeFalse(isConcreteWasmType(curr->ifTrue->type), curr, "if without else must not return a value in body");
181  if (curr->condition->type != unreachable) {
182  shouldBeEqual(curr->type, none, curr, "if without else and reachable condition must be none");
183  }
184  } else {
185  if (curr->type != unreachable) {
186  shouldBeEqualOrFirstIsUnreachable(curr->ifTrue->type, curr->type, curr, "returning if-else's true must have right type");
187  shouldBeEqualOrFirstIsUnreachable(curr->ifFalse->type, curr->type, curr, "returning if-else's false must have right type");
188  } else {
189  if (curr->condition->type != unreachable) {
190  shouldBeEqual(curr->ifTrue->type, unreachable, curr, "unreachable if-else must have unreachable true");
191  shouldBeEqual(curr->ifFalse->type, unreachable, curr, "unreachable if-else must have unreachable false");
192  }
193  }
194  }

197  // override scan to add a pre and a post check task to all nodes
198  static void scan(WasmValidator* self, Expression** currp) {
199  PostWalker<WasmValidator>::scan(self, currp);
200 
201  auto* curr = *currp;
202  if (curr->is<Block>()) self->pushTask(visitPreBlock, currp);
203  if (curr->is<Loop>()) self->pushTask(visitPreLoop, currp);

206  void noteBreak(Name name, Expression* value, Expression* curr) {
207  WasmType valueType = none;
208  Index arity = 0;
209  if (value) {
210  valueType = value->type;
211  shouldBeUnequal(valueType, none, curr, "breaks must have a valid value");
212  arity = 1;
213  }
214  if (!shouldBeTrue(breakTargets[name].size() > 0, curr, "all break targets must be valid")) return;
215  auto* target = breakTargets[name].back();
216  if (breakInfos.count(target) == 0) {
217  breakInfos[target] = BreakInfo(valueType, arity);
218  } else {
219  auto& info = breakInfos[target];
220  if (info.type == unreachable) {
221  info.type = valueType;
222  } else if (valueType != unreachable) {
223  if (valueType != info.type) {
224  info.type = none; // a poison value that must not be consumed
225  }
226  }
227  if (arity != info.arity) {
228  info.arity = Index(-1); // a poison value
229  }
230  }
231  }

234  if (BranchUtils::isBranchTaken(curr)) {
235  noteBreak(curr->name, curr->value, curr);

238  shouldBeTrue(curr->condition->type == unreachable || curr->condition->type == i32, curr, "break condition must be i32");

241  void visitSwitch(Switch *curr) {
242  // note breaks (that are actually taken)
243  if (BranchUtils::isBranchTaken(curr)) {
244  for (auto& target : curr->targets) {
245  noteBreak(target, curr->value, curr);
246  }
247  noteBreak(curr->default_, curr->value, curr);
248  }
249  shouldBeTrue(curr->condition->type == unreachable || curr->condition->type == i32, curr, "br_table condition must be i32");
250  }
251  void visitCall(Call *curr) {
252  if (!validateGlobally) return;
253  auto* target = getModule()->getFunctionOrNull(curr->target);
254  if (!shouldBeTrue(!!target, curr, "call target must exist")) return;
255  if (!shouldBeTrue(curr->operands.size() == target->params.size(), curr, "call param number must match")) return;
256  for (size_t i = 0; i < curr->operands.size(); i++) {
257  if (!shouldBeEqualOrFirstIsUnreachable(curr->operands[i]->type, target->params[i], curr, "call param types must match")) {
258  std::cerr << "(on argument " << i << ")\n";
259  }
260  }
261  }

764  auto oldType = curr->type;
765  ReFinalizeNode().visit(curr);
766  auto newType = curr->type;
