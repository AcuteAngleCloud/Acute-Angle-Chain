7 // The Great Type Loophole (C++14)
8 // Initial implementation by Alexandr Poltavsky, http://alexpolt.github.io
9 //
10 // Description:
11 // The Great Type Loophole is a technique that allows to exchange type information with template
12 // instantiations. Basically you can assign and read type information during compile time.
13 // Here it is used to detect data members of a data type. I described it for the first time in
14 // this blog post http://alexpolt.github.io/type-loophole.html .
15 //
16 // This technique exploits the http://www.open-std.org/jtc1/sc22/wg21/docs/cwg_active.html#2118
17 // CWG 2118. Stateful metaprogramming via friend injection
18 // Note: CWG agreed that such techniques should be ill-formed, although the mechanism for prohibiting them is as yet undetermined.

50 // tag<T,N> generates friend declarations and helps with overload resolution.
51 // There are two types: one with the auto return type, which is the way we read types later.
52 // The second one is used in the detection of instantiations without which we'd get multiple
53 // definitions.
54 
55 template <class T, std::size_t N>
56 struct tag {
57  friend auto loophole(tag<T,N>);
58 };

60 // For returning non default constructible types. Never used at runtime! GCC's std::declval may not be used in potentionally evaluated contexts, so it does not work here.
61 template <class T> constexpr T& unsafe_declval_like() noexcept {
62  T* ptr = nullptr;
63  return *ptr;
64 }

66 // The definitions of friend functions.
67 template <class T, class U, std::size_t N, bool B>
68 struct fn_def {
69  friend auto loophole(tag<T,N>) { return boost::pfr::detail::unsafe_declval_like< std::remove_all_extents_t<U> >(); }

77 // This has a templated conversion operator which in turn triggers instantiations.
78 // Important point, using sizeof seems to be more reliable. Also default template
79 // arguments are "cached" (I think). To fix that I provide a U template parameter to
80 // the ins functions which do the detection using constexpr friend functions and SFINAE.
81 template <class T, std::size_t N>
82 struct loophole_ubiq {
83  template<class U, std::size_t M> static std::size_t ins(...);
84  template<class U, std::size_t M, std::size_t = sizeof(loophole(tag<T,M>{})) > static char ins(int);
85 
86  template<class U, std::size_t = sizeof(fn_def<T, U, N, sizeof(ins<U, N>(0)) == sizeof(char)>)>
87  constexpr operator U&() const noexcept; // `const` here helps to avoid ambiguity in loophole instantiations. optional_like test validate that behavior.
88 };

104 template <class T>
105 auto tie_as_tuple_loophole_impl(T& lvalue) noexcept {
106  using type = std::remove_cv_t<std::remove_reference_t<T>>;
107  using indexes = std::make_index_sequence<fields_count<type>()>;
108  using tuple_type = typename loophole_type_list<type, indexes>::type;
109 
110  return boost::pfr::detail::make_flat_tuple_of_references(
111  lvalue,
112  offset_based_getter<type, tuple_type>{},
113  size_t_<0>{},
114  size_t_<tuple_type::size_v>{}
115  );
116 }
