32  std::vector<uint8_t> parse_initial_memory(const Module& module) {
33  std::vector<uint8_t> mem_image;
34 
35  for(const DataSegment& data_segment : module.dataSegments) {
36  FC_ASSERT(data_segment.baseOffset.type == InitializerExpression::Type::i32_const);
37  FC_ASSERT(module.memories.defs.size());
38  const U32 base_offset = data_segment.baseOffset.i32;
39  const Uptr memory_size = (module.memories.defs[0].type.size.min << IR::numBytesPerPageLog2);
40  if(base_offset >= memory_size || base_offset + data_segment.data.size() > memory_size)
41  FC_THROW_EXCEPTION(wasm_execution_error, "WASM data segment outside of valid memory range");
42  if(base_offset + data_segment.data.size() > mem_image.size())
43  mem_image.resize(base_offset + data_segment.data.size(), 0x00);
44  memcpy(mem_image.data() + base_offset, data_segment.data.data(), data_segment.data.size());
45  }
46 
47  return mem_image;
48  }
