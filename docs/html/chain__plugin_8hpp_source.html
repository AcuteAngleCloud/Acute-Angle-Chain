209  template <typename IndexType, typename Scope>
210  read_only::get_table_rows_result get_table_rows_ex( const read_only::get_table_rows_params& p, const abi_def& abi )const {
211  read_only::get_table_rows_result result;
212  const auto& d = db.get_database();
213 
214  uint64_t scope = 0;
215  try {
216  name s(p.scope);
217  scope = s.value;
218  } catch( ... ) {
219  try {
220  auto trimmed_scope_str = p.scope;
221  boost::trim(trimmed_scope_str);
222  scope = boost::lexical_cast<uint64_t>(trimmed_scope_str.c_str(), trimmed_scope_str.size());
223  } catch( ... ) {
224  try {
225  auto symb = aacio::chain::symbol::from_string(p.scope);
226  scope = symb.value();
227  } catch( ... ) {
228  try {
229  scope = ( aacio::chain::string_to_symbol( 0, p.scope.c_str() ) >> 8 );
230  } catch( ... ) {
231  FC_ASSERT( false, "could not convert scope string to any of the following: uint64_t, valid name, or valid symbol (with or without the precision)" );
232  }
233  }
234  }
235  }
236 
237  abi_serializer abis;
238  abis.set_abi(abi);
239  const auto* t_id = d.find<chain::contracts::table_id_object, chain::contracts::by_code_scope_table>(boost::make_tuple(p.code, scope, p.table));
240  if (t_id != nullptr) {
241  const auto &idx = d.get_index<IndexType, Scope>();
242  decltype(t_id->id) next_tid(t_id->id._id + 1);
243  auto lower = idx.lower_bound(boost::make_tuple(t_id->id));
244  auto upper = idx.lower_bound(boost::make_tuple(next_tid));
245 
246  if (p.lower_bound.size()) {
247  lower = idx.lower_bound(boost::make_tuple(t_id->id, fc::variant(
248  p.lower_bound).as<typename IndexType::value_type::key_type>()));
249  }
250  if (p.upper_bound.size()) {
251  upper = idx.lower_bound(boost::make_tuple(t_id->id, fc::variant(
252  p.upper_bound).as<typename IndexType::value_type::key_type>()));
253  }
254 
255  vector<char> data;
256 
257  auto end = fc::time_point::now() + fc::microseconds(1000 * 10);
258 
259  unsigned int count = 0;
260  auto itr = lower;
261  for (itr = lower; itr != upper; ++itr) {
262  copy_inline_row(*itr, data);
263 
264  if (p.json) {
265  result.rows.emplace_back(abis.binary_to_variant(abis.get_table_type(p.table), data));
266  } else {
267  result.rows.emplace_back(fc::variant(data));
268  }
269 
270  if (++count == p.limit || fc::time_point::now() > end) {
271  break;
272  }
273  }
274  if (itr != upper) {
275  result.more = true;
276  }
277  }
