187  template<typename Stream> inline void unpack( Stream& s, signed_int& vi ) {
188  uint32_t v = 0; char b = 0; int by = 0;
189  do {
190  s.get(b);
191  v |= uint32_t(uint8_t(b) & 0x7f) << by;
192  by += 7;
193  } while( uint8_t(b) & 0x80 );
194  vi.value = ((v>>1) ^ (v>>31)) + (v&0x01);
195  vi.value = v&0x01 ? vi.value : -vi.value;
196  vi.value = -vi.value;
