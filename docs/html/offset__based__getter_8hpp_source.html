23 // Our own implementation of std::aligned_storage. On godbolt with MSVC, I have compilation errors
24 // using the standard version, it seems the compiler cannot generate default ctor.

26 template<std::size_t s, std::size_t a>
27 struct internal_aligned_storage {
28  alignas(a) char storage_[s];
29 };

31 // Metafunction that replaces tuple<T1, T2, T3, ...> with
32 // tuple<std::aligned_storage_t<sizeof(T1), alignof(T1)>, std::aligned_storage<sizeof(T2), alignof(T2)>, ...>
33 //
34 // The point is, the new tuple is "layout compatible" in the sense that members have the same offsets,
35 // but this tuple is constexpr constructible.
36 
37 template <typename T>
38 struct tuple_of_aligned_storage;
39 
40 template <typename... Ts>
41 struct tuple_of_aligned_storage<sequence_tuple::tuple<Ts...>> {
42  using type = sequence_tuple::tuple<internal_aligned_storage<sizeof(Ts), alignof(Ts)>...>;
43 };
44 
45 // Note: If pfr has a typelist also, could also have an overload for that here
46 
47 template <typename T>
48 using tuple_of_aligned_storage_t = typename tuple_of_aligned_storage<T>::type;

50 /***
51  * Given a structure type and its sequence of members, we want to build a function
52  * object "getter" that implements a version of `std::get` using offset arithmetic
53  * and reinterpret_cast.
54  *
55  * typename U should be a user-defined struct
56  * typename S should be a sequence_tuple which is layout compatible with U
57  */
58 
59 template <typename U, typename S>
60 class offset_based_getter {
61  using this_t = offset_based_getter<U, S>;
62 
63  static_assert(sizeof(U) == sizeof(S), "Member sequence does not indicate correct size for struct type!");
64  static_assert(alignof(U) == alignof(S), "Member sequence does not indicate correct alignment for struct type!");
65 
66  static_assert(!std::is_const<U>::value, "const should be stripped from user-defined type when using offset_based_getter or overload resolution will be ambiguous later, this indicates an error within pfr");
67  static_assert(!std::is_reference<U>::value, "reference should be stripped from user-defined type when using offset_based_getter or overload resolution will be ambiguous later, this indicates an error within pfr");
68  static_assert(!std::is_volatile<U>::value, "volatile should be stripped from user-defined type when using offset_based_getter or overload resolution will be ambiguous later. this indicates an error within pfr");
69 
70  // Get type of idx'th member
71  template <std::size_t idx>
72  using index_t = typename sequence_tuple::tuple_element<idx, S>::type;
73 
74  // Get offset of idx'th member
75  // Idea: Layout object has the same offsets as instance of S, so if S and U are layout compatible, then these offset
76  // calculations are correct.
77  template <std::size_t idx>
78  static constexpr std::ptrdiff_t offset() noexcept {
79  constexpr tuple_of_aligned_storage_t<S> layout{};
80  return &sequence_tuple::get<idx>(layout).storage_[0] - &sequence_tuple::get<0>(layout).storage_[0];
81  }
82 
83  // Encapsulates offset arithmetic and reinterpret_cast
84  template <std::size_t idx>
85  static index_t<idx> * get_pointer(U * u) noexcept {
86  return reinterpret_cast<index_t<idx> *>(reinterpret_cast<char *>(u) + this_t::offset<idx>());
87  }
88 
89  template <std::size_t idx>
90  static const index_t<idx> * get_pointer(const U * u) noexcept {
91  return reinterpret_cast<const index_t<idx> *>(reinterpret_cast<const char *>(u) + this_t::offset<idx>());
92  }
93 
94  template <std::size_t idx>
95  static volatile index_t<idx> * get_pointer(volatile U * u) noexcept {
96  return reinterpret_cast<volatile index_t<idx> *>(reinterpret_cast<volatile char *>(u) + this_t::offset<idx>());
97  }
98 
99  template <std::size_t idx>
100  static const volatile index_t<idx> * get_pointer(const volatile U * u) noexcept {
101  return reinterpret_cast<const volatile index_t<idx> *>(reinterpret_cast<const volatile char *>(u) + this_t::offset<idx>());
102  }
103 
104 public:
105  template <std::size_t idx>
106  index_t<idx> & get(U & u, size_t_<idx>) const noexcept {
107  return *this_t::get_pointer<idx>(std::addressof(u));
108  }
109 
110  template <std::size_t idx>
111  index_t<idx> const & get(U const & u, size_t_<idx>) const noexcept {
112  return *this_t::get_pointer<idx>(std::addressof(u));
113  }
114 
115  template <std::size_t idx>
116  index_t<idx> volatile & get(U volatile & u, size_t_<idx>) const noexcept {
117  return *this_t::get_pointer<idx>(std::addressof(u));
118  }
119 
120  template <std::size_t idx>
121  index_t<idx> const volatile & get(U const volatile & u, size_t_<idx>) const noexcept {
122  return *this_t::get_pointer<idx>(std::addressof(u));
123  }
124 
125  // rvalues must no be used here, to avoid template instantiation bloats.
126  template <std::size_t idx>
127  index_t<idx> && get(rvalue_t<U> u, size_t_<idx>) const = delete;
128 };
