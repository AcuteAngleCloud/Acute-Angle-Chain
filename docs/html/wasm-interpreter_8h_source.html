52 // Stuff that flows around during executing expressions: a literal, or a change in control flow.
53 class Flow {
54 public:
55  Flow() {}
56  Flow(Literal value) : value(value) {}
57  Flow(Name breakTo) : breakTo(breakTo) {}
58 
59  Literal value;
60  Name breakTo; // if non-null, a break is going on
61 
62  bool breaking() { return breakTo.is(); }
63 
64  void clearIf(Name target) {
65  if (breakTo == target) {
66  breakTo.clear();
67  }
68  }
69 
70  friend std::ostream& operator<<(std::ostream& o, Flow& flow) {
71  o << "(flow " << (flow.breakTo.is() ? flow.breakTo.str : "-") << " : " << flow.value << ')';
72  return o;
73  }
74 };

124  Flow visitBlock(Block *curr) {
125  NOTE_ENTER("Block");
126  // special-case Block, because Block nesting (in their first element) can be incredibly deep
127  std::vector<Block*> stack;
128  stack.push_back(curr);
129  while (curr->list.size() > 0 && curr->list[0]->is<Block>()) {
130  curr = curr->list[0]->cast<Block>();
131  stack.push_back(curr);
132  }
133  Flow flow;
134  auto* top = stack.back();
135  while (stack.size() > 0) {
136  curr = stack.back();
137  stack.pop_back();
138  if (flow.breaking()) {
139  flow.clearIf(curr->name);
140  continue;
141  }
142  auto& list = curr->list;
143  for (size_t i = 0; i < list.size(); i++) {
144  if (curr != top && i == 0) {
145  // one of the block recursions we already handled
146  continue;
147  }
148  flow = visit(list[i]);
149  if (flow.breaking()) {
150  flow.clearIf(curr->name);
151  break;
152  }
153  }
154  }
155  return flow;
156  }
157  Flow visitIf(If *curr) {
158  NOTE_ENTER("If");
159  Flow flow = visit(curr->condition);
160  if (flow.breaking()) return flow;
161  NOTE_EVAL1(flow.value);
162  if (flow.value.geti32()) {
163  Flow flow = visit(curr->ifTrue);
164  if (!flow.breaking() && !curr->ifFalse) flow.value = Literal(); // if_else returns a value, but if does not
165  return flow;
166  }
167  if (curr->ifFalse) return visit(curr->ifFalse);
168  return Flow();
169  }
170  Flow visitLoop(Loop *curr) {
171  NOTE_ENTER("Loop");
172  while (1) {
173  Flow flow = visit(curr->body);
174  if (flow.breaking()) {
175  if (flow.breakTo == curr->name) continue; // lol
176  }
177  return flow; // loop does not loop automatically, only continue achieves that
178  }
179  }
180  Flow visitBreak(Break *curr) {
181  NOTE_ENTER("Break");
182  bool condition = true;
183  Flow flow;
184  if (curr->value) {
185  flow = visit(curr->value);
186  if (flow.breaking()) return flow;
187  }
188  if (curr->condition) {
189  Flow conditionFlow = visit(curr->condition);
190  if (conditionFlow.breaking()) return conditionFlow;
191  condition = conditionFlow.value.getInteger() != 0;
192  if (!condition) return flow;
193  }
194  flow.breakTo = curr->name;
195  return flow;
196  }
197  Flow visitSwitch(Switch *curr) {
198  NOTE_ENTER("Switch");
199  Flow flow;
200  Literal value;
201  if (curr->value) {
202  flow = visit(curr->value);
203  if (flow.breaking()) return flow;
204  value = flow.value;
205  NOTE_EVAL1(value);
206  }
207  flow = visit(curr->condition);
208  if (flow.breaking()) return flow;
209  int64_t index = flow.value.getInteger();
210  Name target = curr->default_;
211  if (index >= 0 && (size_t)index < curr->targets.size()) {
212  target = curr->targets[(size_t)index];
213  }
214  flow.breakTo = target;
215  flow.value = value;
216  return flow;
217  }

224  Flow visitUnary(Unary *curr) {
225  NOTE_ENTER("Unary");
226  Flow flow = visit(curr->value);
227  if (flow.breaking()) return flow;
228  Literal value = flow.value;
229  NOTE_EVAL1(value);
230  if (value.type == i32) {
231  switch (curr->op) {
232  case ClzInt32: return value.countLeadingZeroes();
233  case CtzInt32: return value.countTrailingZeroes();
234  case PopcntInt32: return value.popCount();
235  case EqZInt32: return Literal(int32_t(value == Literal(int32_t(0))));
236  case ReinterpretInt32: return value.castToF32();
237  case ExtendSInt32: return value.extendToSI64();
238  case ExtendUInt32: return value.extendToUI64();
239  case ConvertUInt32ToFloat32: return value.convertUToF32();
240  case ConvertUInt32ToFloat64: return value.convertUToF64();
241  case ConvertSInt32ToFloat32: return value.convertSToF32();
242  case ConvertSInt32ToFloat64: return value.convertSToF64();
243  default: WASM_UNREACHABLE();
244  }
245  }
246  if (value.type == i64) {
247  switch (curr->op) {
248  case ClzInt64: return value.countLeadingZeroes();
249  case CtzInt64: return value.countTrailingZeroes();
250  case PopcntInt64: return value.popCount();
251  case EqZInt64: return Literal(int32_t(value == Literal(int64_t(0))));
252  case WrapInt64: return value.truncateToI32();
253  case ReinterpretInt64: return value.castToF64();
254  case ConvertUInt64ToFloat32: return value.convertUToF32();
255  case ConvertUInt64ToFloat64: return value.convertUToF64();
256  case ConvertSInt64ToFloat32: return value.convertSToF32();
257  case ConvertSInt64ToFloat64: return value.convertSToF64();
258  default: WASM_UNREACHABLE();
259  }
260  }
261  if (value.type == f32) {
262  switch (curr->op) {
263  case NegFloat32: return value.neg();
264  case AbsFloat32: return value.abs();
265  case CeilFloat32: return value.ceil();
266  case FloorFloat32: return value.floor();
267  case TruncFloat32: return value.trunc();
268  case NearestFloat32: return value.nearbyint();
269  case SqrtFloat32: return value.sqrt();
270  case TruncSFloat32ToInt32:
271  case TruncSFloat32ToInt64: return truncSFloat(curr, value);
272  case TruncUFloat32ToInt32:
273  case TruncUFloat32ToInt64: return truncUFloat(curr, value);
274  case ReinterpretFloat32: return value.castToI32();
275  case PromoteFloat32: return value.extendToF64();
276  default: WASM_UNREACHABLE();
277  }
278  }
279  if (value.type == f64) {
280  switch (curr->op) {
281  case NegFloat64: return value.neg();
282  case AbsFloat64: return value.abs();
283  case CeilFloat64: return value.ceil();
284  case FloorFloat64: return value.floor();
285  case TruncFloat64: return value.trunc();
286  case NearestFloat64: return value.nearbyint();
287  case SqrtFloat64: return value.sqrt();
288  case TruncSFloat64ToInt32:
289  case TruncSFloat64ToInt64: return truncSFloat(curr, value);
290  case TruncUFloat64ToInt32:
291  case TruncUFloat64ToInt64: return truncUFloat(curr, value);
292  case ReinterpretFloat64: return value.castToI64();
293  case DemoteFloat64: {
294  double val = value.getFloat();
295  if (std::isnan(val)) return Literal(float(val));
296  if (std::isinf(val)) return Literal(float(val));
297  // when close to the limit, but still truncatable to a valid value, do that
298  // see https://github.com/WebAssembly/sexpr-wasm-prototype/blob/2d375e8d502327e814d62a08f22da9d9b6b675dc/src/wasm-interpreter.c#L247
299  uint64_t bits = value.reinterpreti64();
300  if (bits > 0x47efffffe0000000ULL && bits < 0x47effffff0000000ULL) return Literal(std::numeric_limits<float>::max());
301  if (bits > 0xc7efffffe0000000ULL && bits < 0xc7effffff0000000ULL) return Literal(-std::numeric_limits<float>::max());
302  // when we must convert to infinity, do that
303  if (val < -std::numeric_limits<float>::max()) return Literal(-std::numeric_limits<float>::infinity());
304  if (val > std::numeric_limits<float>::max()) return Literal(std::numeric_limits<float>::infinity());
305  return value.truncateToF32();
306  }
307  default: WASM_UNREACHABLE();
308  }
309  }
310  WASM_UNREACHABLE();
311  }
312  Flow visitBinary(Binary *curr) {
313  NOTE_ENTER("Binary");
314  Flow flow = visit(curr->left);
315  if (flow.breaking()) return flow;
316  Literal left = flow.value;
317  flow = visit(curr->right);
318  if (flow.breaking()) return flow;
319  Literal right = flow.value;
320  NOTE_EVAL2(left, right);
321  assert(isConcreteWasmType(curr->left->type) ? left.type == curr->left->type : true);
322  assert(isConcreteWasmType(curr->right->type) ? right.type == curr->right->type : true);
323  if (left.type == i32) {
324  switch (curr->op) {
325  case AddInt32: return left.add(right);
326  case SubInt32: return left.sub(right);
327  case MulInt32: return left.mul(right);
328  case DivSInt32: {
329  if (right.getInteger() == 0) trap("i32.div_s by 0");
330  if (left.getInteger() == std::numeric_limits<int32_t>::min() && right.getInteger() == -1) trap("i32.div_s overflow"); // signed division overflow
331  return left.divS(right);
332  }
333  case DivUInt32: {
334  if (right.getInteger() == 0) trap("i32.div_u by 0");
335  return left.divU(right);
336  }
337  case RemSInt32: {
338  if (right.getInteger() == 0) trap("i32.rem_s by 0");
339  if (left.getInteger() == std::numeric_limits<int32_t>::min() && right.getInteger() == -1) return Literal(int32_t(0));
340  return left.remS(right);
341  }
342  case RemUInt32: {
343  if (right.getInteger() == 0) trap("i32.rem_u by 0");
344  return left.remU(right);
345  }
346  case AndInt32: return left.and_(right);
347  case OrInt32: return left.or_(right);
348  case XorInt32: return left.xor_(right);
349  case ShlInt32: return left.shl(right.and_(Literal(int32_t(31))));
350  case ShrUInt32: return left.shrU(right.and_(Literal(int32_t(31))));
351  case ShrSInt32: return left.shrS(right.and_(Literal(int32_t(31))));
352  case RotLInt32: return left.rotL(right);
353  case RotRInt32: return left.rotR(right);
354  case EqInt32: return left.eq(right);
355  case NeInt32: return left.ne(right);
356  case LtSInt32: return left.ltS(right);
357  case LtUInt32: return left.ltU(right);
358  case LeSInt32: return left.leS(right);
359  case LeUInt32: return left.leU(right);
360  case GtSInt32: return left.gtS(right);
361  case GtUInt32: return left.gtU(right);
362  case GeSInt32: return left.geS(right);
363  case GeUInt32: return left.geU(right);
364  default: WASM_UNREACHABLE();
365  }
366  } else if (left.type == i64) {
367  switch (curr->op) {
368  case AddInt64: return left.add(right);
369  case SubInt64: return left.sub(right);
370  case MulInt64: return left.mul(right);
371  case DivSInt64: {
372  if (right.getInteger() == 0) trap("i64.div_s by 0");
373  if (left.getInteger() == LLONG_MIN && right.getInteger() == -1LL) trap("i64.div_s overflow"); // signed division overflow
374  return left.divS(right);
375  }
376  case DivUInt64: {
377  if (right.getInteger() == 0) trap("i64.div_u by 0");
378  return left.divU(right);
379  }
380  case RemSInt64: {
381  if (right.getInteger() == 0) trap("i64.rem_s by 0");
382  if (left.getInteger() == LLONG_MIN && right.getInteger() == -1LL) return Literal(int64_t(0));
383  return left.remS(right);
384  }
385  case RemUInt64: {
386  if (right.getInteger() == 0) trap("i64.rem_u by 0");
387  return left.remU(right);
388  }
389  case AndInt64: return left.and_(right);
390  case OrInt64: return left.or_(right);
391  case XorInt64: return left.xor_(right);
392  case ShlInt64: return left.shl(right.and_(Literal(int64_t(63))));
393  case ShrUInt64: return left.shrU(right.and_(Literal(int64_t(63))));
394  case ShrSInt64: return left.shrS(right.and_(Literal(int64_t(63))));
395  case RotLInt64: return left.rotL(right);
396  case RotRInt64: return left.rotR(right);
397  case EqInt64: return left.eq(right);
398  case NeInt64: return left.ne(right);
399  case LtSInt64: return left.ltS(right);
400  case LtUInt64: return left.ltU(right);
401  case LeSInt64: return left.leS(right);
402  case LeUInt64: return left.leU(right);
403  case GtSInt64: return left.gtS(right);
404  case GtUInt64: return left.gtU(right);
405  case GeSInt64: return left.geS(right);
406  case GeUInt64: return left.geU(right);
407  default: WASM_UNREACHABLE();
408  }
409  } else if (left.type == f32 || left.type == f64) {
410  switch (curr->op) {
411  case AddFloat32: case AddFloat64: return left.add(right);
412  case SubFloat32: case SubFloat64: return left.sub(right);
413  case MulFloat32: case MulFloat64: return left.mul(right);
414  case DivFloat32: case DivFloat64: return left.div(right);
415  case CopySignFloat32: case CopySignFloat64: return left.copysign(right);
416  case MinFloat32: case MinFloat64: return left.min(right);
417  case MaxFloat32: case MaxFloat64: return left.max(right);
418  case EqFloat32: case EqFloat64: return left.eq(right);
419  case NeFloat32: case NeFloat64: return left.ne(right);
420  case LtFloat32: case LtFloat64: return left.lt(right);
421  case LeFloat32: case LeFloat64: return left.le(right);
422  case GtFloat32: case GtFloat64: return left.gt(right);
423  case GeFloat32: case GeFloat64: return left.ge(right);
424  default: WASM_UNREACHABLE();
425  }
426  }
427  WASM_UNREACHABLE();
428  }

467  Literal truncSFloat(Unary* curr, Literal value) {
468  double val = value.getFloat();
469  if (std::isnan(val)) trap("truncSFloat of nan");
470  if (curr->type == i32) {
471  if (value.type == f32) {
472  if (!isInRangeI32TruncS(value.reinterpreti32())) trap("i32.truncSFloat overflow");
473  } else {
474  if (!isInRangeI32TruncS(value.reinterpreti64())) trap("i32.truncSFloat overflow");
475  }
476  return Literal(int32_t(val));
477  } else {
478  if (value.type == f32) {
479  if (!isInRangeI64TruncS(value.reinterpreti32())) trap("i64.truncSFloat overflow");
480  } else {
481  if (!isInRangeI64TruncS(value.reinterpreti64())) trap("i64.truncSFloat overflow");
482  }
483  return Literal(int64_t(val));
484  }
485  }

487  Literal truncUFloat(Unary* curr, Literal value) {
488  double val = value.getFloat();
489  if (std::isnan(val)) trap("truncUFloat of nan");
490  if (curr->type == i32) {
491  if (value.type == f32) {
492  if (!isInRangeI32TruncU(value.reinterpreti32())) trap("i32.truncUFloat overflow");
493  } else {
494  if (!isInRangeI32TruncU(value.reinterpreti64())) trap("i32.truncUFloat overflow");
495  }
496  return Literal(uint32_t(val));
497  } else {
498  if (value.type == f32) {
499  if (!isInRangeI64TruncU(value.reinterpreti32())) trap("i64.truncUFloat overflow");
500  } else {
501  if (!isInRangeI64TruncU(value.reinterpreti64())) trap("i64.truncUFloat overflow");
502  }
503  return Literal(uint64_t(val));

532 //
533 // An instance of a WebAssembly module, which can execute it via AST interpretation.
534 //
535 // To embed this interpreter, you need to provide an ExternalInterface instance
536 // (see below) which provides the embedding-specific details, that is, how to
537 // connect to the embedding implementation.
538 //
539 // To call into the interpreter, use callExport.
540 //
541 
542 template<typename GlobalManager, typename SubType>
543 class ModuleInstanceBase {
544 public:
545  //
546  // You need to implement one of these to create a concrete interpreter. The
547  // ExternalInterface provides embedding-specific functionality like calling
548  // an imported function or accessing memory.
549  //
550  struct ExternalInterface {
551  virtual void init(Module& wasm, SubType& instance) {}
552  virtual void importGlobals(GlobalManager& globals, Module& wasm) = 0;
553  virtual Literal callImport(Import* import, LiteralList& arguments) = 0;
554  virtual Literal callTable(Index index, LiteralList& arguments, WasmType result, SubType& instance) = 0;
555  virtual void growMemory(Address oldSize, Address newSize) = 0;
556  virtual void trap(const char* why) = 0;
557 
558  // the default impls for load and store switch on the sizes. you can either
559  // customize load/store, or the sub-functions which they call
560  virtual Literal load(Load* load, Address addr) {
561  switch (load->type) {
562  case i32: {
563  switch (load->bytes) {
564  case 1: return load->signed_ ? Literal((int32_t)load8s(addr)) : Literal((int32_t)load8u(addr));
565  case 2: return load->signed_ ? Literal((int32_t)load16s(addr)) : Literal((int32_t)load16u(addr));
566  case 4: return load->signed_ ? Literal((int32_t)load32s(addr)) : Literal((int32_t)load32u(addr));
567  default: WASM_UNREACHABLE();
568  }
569  break;
570  }
571  case i64: {
572  switch (load->bytes) {
573  case 1: return load->signed_ ? Literal((int64_t)load8s(addr)) : Literal((int64_t)load8u(addr));
574  case 2: return load->signed_ ? Literal((int64_t)load16s(addr)) : Literal((int64_t)load16u(addr));
575  case 4: return load->signed_ ? Literal((int64_t)load32s(addr)) : Literal((int64_t)load32u(addr));
576  case 8: return load->signed_ ? Literal((int64_t)load64s(addr)) : Literal((int64_t)load64u(addr));
